 51 /**
 52  * Create an interactor for a given element in the DOM tree.
 53  * 
 54  * @constructor
 55  * @param {Element} element The DOM element to be observed.
 56  * @extends X.base
 57  */
 58 X.interactor = function(element) {
 59 
 60   // check if we have a valid element
 61   if (!goog.isDefAndNotNull(element) || !(element instanceof Element)) {
 62     
 63     throw new Error('Could not add interactor to the given element.');
 64     
 65   }
 66   
 67   //
 68   // call the standard constructor of X.base
 69   goog.base(this);
 70   
 71   //
 72   // class attributes
 73   
 74   /**
 75    * @inheritDoc
 76    * @const
 77    */
 78   this._classname = 'interactor';
 79   
 80   /**
 81    * The observed DOM element of this interactor.
 82    * 
 83    * @type {!Element}
 84    * @protected
 85    */
 86   this._element = element;
 87   
 88   /**
 89    * The listener id for mouse wheel observation.
 90    * 
 91    * @type {?number}
 92    * @protected
 93    */
 94   this._mouseWheelListener = null;
 95   
 96   /**
 97    * The listener id for mouse down observation.
 98    * 
 99    * @type {?number}
100    * @protected
101    */
102   this._mouseDownListener = null;
103   
104   /**
105    * The listener id for mouse up observation.
106    * 
107    * @type {?number}
108    * @protected
109    */
110   this._mouseUpListener = null;
111   
112   /**
113    * The listener id for mouse move observation.
114    * 
115    * @type {?number}
116    * @protected
117    */
118   this._mouseMoveListener = null;
119   
120   /**
121    * The listener id for mouse out observation.
122    * 
123    * @type {?number}
124    * @protected
125    */
126   this._mouseOutListener = null;
127   
128   /**
129    * The browser independent mouse wheel handler.
130    * 
131    * @type {?goog.events.MouseWheelHandler}
132    * @protected
133    */
134   this._mouseWheelHandler = null;
135   
136   /**
137    * Indicates if the mouse is inside the element.
138    * 
139    * @type {boolean}
140    * @protected
141    */
142   this._mouseInside = true;
143   
144   /**
145    * Indicates if the left mouse button is pressed.
146    * 
147    * @type {boolean}
148    * @protected
149    */
150   this._leftButtonDown = false;
151   
152   /**
153    * Indicates if the middle mouse button is pressed.
154    * 
155    * @type {boolean}
156    * @protected
157    */
158   this._middleButtonDown = false;
159   
160   /**
161    * Indicates if the right mouse button is pressed.
162    * 
163    * @type {boolean}
164    * @protected
165    */
166   this._rightButtonDown = false;
167   
168   /**
169    * The current mouse position.
170    * 
171    * @type {!Array}
172    * @protected
173    */
174   this._mousePosition = [0, 0];
175   
176   /**
177    * The previous mouse position.
178    * 
179    * @type {!goog.math.Vec2}
180    * @protected
181    */
182   this._lastMousePosition = new goog.math.Vec2(0, 0);
183   
184   /**
185    * The configuration of this interactor.
186    * 
187    * @enum {boolean}
188    */
189   this['config'] = {
190     'MOUSEWHEEL_ENABLED': true,
191     'MOUSECLICKS_ENABLED': true,
192     'KEYBOARD_ENABLED': true,
193     'HOVERING_ENABLED': true,
194     'CONTEXTMENU_ENABLED': false
195   };
196   
197 };
198 // inherit from X.base
199 goog.inherits(X.interactor, X.base);

238 /**
239  * Observe mouse wheel interaction on the associated DOM element.
240  */
241 X.interactor.prototype.init = function() {
242 
243   if (this['config']['MOUSEWHEEL_ENABLED']) {
244     
245     // we use the goog.events.MouseWheelHandler for a browser-independent
246     // implementation
247     this._mouseWheelHandler = new goog.events.MouseWheelHandler(this._element);
248     
249     this._mouseWheelListener = goog.events.listen(this._mouseWheelHandler,
250         goog.events.MouseWheelHandler.EventType.MOUSEWHEEL, this.onMouseWheel_
251             .bind(this));
252     
253   } else {
254     
255     // remove all mouse wheel observers, if they exist..
256     goog.events.unlistenByKey(this._mouseWheelListener);
257     
258     this._mouseWheelHandler = null;
259     
260   }
261   
262   if (this['config']['MOUSECLICKS_ENABLED']) {
263     
264     // mouse down
265     this._mouseDownListener = goog.events.listen(this._element,
266         goog.events.EventType.MOUSEDOWN, this.onMouseDown_.bind(this));
267     
268     // mouse up
269     this._mouseUpListener = goog.events.listen(this._element,
270         goog.events.EventType.MOUSEUP, this.onMouseUp_.bind(this));
271     
272   } else {
273     
274     // remove the observer, if it exists..
275     // goog.events.unlisten(this._element, goog.events.EventType.MOUSEDOWN);
276     goog.events.unlistenByKey(this._mouseDownListener);
277     
278     // remove the observer, if it exists..
279     goog.events.unlistenByKey(this._mouseUpListener);
280     
281   }
282   
283   if (!this['config']['CONTEXTMENU_ENABLED']) {
284     
285     // deactivate right-click context menu
286     // found no way to use goog.events for that? tried everything..
287     // according to http://help.dottoro.com/ljhwjsss.php, this method is
288     // compatible with all browsers but opera
289     this._element.oncontextmenu = function() {
290 
291       return false;
292       
293     };
294     
295   } else {
296     
297     // re-activate right-click context menu
298     this._element.oncontextmenu = null;
299   }
300   
301   if (this['config']['KEYBOARD_ENABLED']) {
302     
303     // the google closure way did not work, so let's do it this way..
304     window.onkeydown = this.onKey_.bind(this);
305     
306   } else {
307     
308     // remove the keyboard observer
309     window.onkeydown = null;
310     
311   }
312   
313   //
314   // we always listen to mouse move events since they are essential for the
315   // other events
316   // we do make sure, we add them only once
317   
318   // remove the observer, if it exists..
319   goog.events.unlistenByKey(this._mouseMoveListener);
320   
321   // remove the observer, if it exists..
322   goog.events.unlistenByKey(this._mouseOutListener);
323   
324   // mouse movement inside the element
325   this._mouseMoveListener = goog.events.listen(this._element,
326       goog.events.EventType.MOUSEMOVE, this.onMouseMovementInside_.bind(this));
327   
328   // mouse movement outside the element
329   this._mouseOutListener = goog.events.listen(this._element,
330       goog.events.EventType.MOUSEOUT, this.onMouseMovementOutside_.bind(this));
331   
332 };

335 /**
336  * Callback for mouse down events on the associated DOM element.
337  * 
338  * @param {Event} event The browser fired event.
339  * @protected
340  */
341 X.interactor.prototype.onMouseDown_ = function(event) {
342 
343   if (event.button == goog.events.BrowserEvent.MouseButton.LEFT) {
344     
345     // left button click
346     this._leftButtonDown = true;
347     
348   } else if (event.button == goog.events.BrowserEvent.MouseButton.MIDDLE) {
349     
350     // middle button click
351     this._middleButtonDown = true;
352     
353   } else if (event.button == goog.events.BrowserEvent.MouseButton.RIGHT) {
354     
355     // right button click
356     this._rightButtonDown = true;
357     
358   }
359   
360   eval("this.onMouseDown(" + this._leftButtonDown + "," +
361       this._middleButtonDown + "," + this._rightButtonDown + ")");
362   
363   // end all hovering since the scene can change and a caption might be
364   // misplaced etc.
365   this.hoverEnd_();
366   
367   // prevent further handling by the browser
368   event.preventDefault();
369   
370 };

451 /**
452  * Callback for mouse movement events outside the associated DOM element. This
453  * resets all internal interactor flags.
454  * 
455  * @param {Event} event The browser fired event.
456  * @protected
457  */
458 X.interactor.prototype.onMouseMovementOutside_ = function(event) {
459 
460   // reset the click flags
461   this._mouseInside = false;
462   if (this['config']['KEYBOARD_ENABLED']) {
463     
464     // if we observe the keyboard, remove the observer here
465     // this is necessary if there are more than one renderer in the document
466     window.onkeydown = null;
467     
468   }
469   
470   this._leftButtonDown = false;
471   this._middleButtonDown = false;
472   this._rightButtonDown = false;
473   // end all hovering since the scene can change and a caption might be
474   // misplaced etc.
475   this.hoverEnd_();
476   this._lastMousePosition = new goog.math.Vec2(0, 0);
477   
478   // prevent further handling by the browser
479   event.preventDefault();
480   
481 };

496 /**
497  * Callback for mouse movement events inside the associated DOM element. This
498  * distinguishes by pressed mouse buttons, key accelerators etc. and fires
499  * proper X.event events.
500  * 
501  * @param {Event} event The browser fired event.
502  * @protected
503  */
504 X.interactor.prototype.onMouseMovementInside_ = function(event) {
505 
506   this['mousemoveEvent'] = event; // we need to buffer the event to run eval in
507   // advanced compilation
508   eval("this.onMouseMove(this['mousemoveEvent'])");
509   
510   this._mouseInside = true;
511   
512   if (this['config']['KEYBOARD_ENABLED'] && window.onkeydown == null) {
513     
514     // we re-gained the focus, enable the keyboard observer again!
515     window.onkeydown = this.onKey_.bind(this);
516     
517 
518   }
519   
520   // prevent any other actions by the browser (f.e. scrolling, selection..)
521   event.preventDefault();
522   
523   // is shift down?
524   var shiftDown = event.shiftKey;
525   
526   // grab the current mouse position
527   this._mousePosition = [event.offsetX, event.offsetY];
528   var currentMousePosition = new goog.math.Vec2(this._mousePosition[0],
529       this._mousePosition[1]);
530   
531   // get the distance in terms of the last mouse move event
532   var distance = this._lastMousePosition.subtract(currentMousePosition);
533   
534   // save the current mouse position as the last one
535   this._lastMousePosition = currentMousePosition.clone();
536   
537   // 
538   // hovering, if enabled..
539   //
540   if (this['config']['HOVERING_ENABLED']) {
541     
542     if (Math.abs(distance.x) > 0 || Math.abs(distance.y) > 0 ||
543         this._middleButtonDown || this._leftButtonDown || this._rightButtonDown) {
544       
545       // there was some mouse movement, let's cancel the hovering countdown
546       this.hoverEnd_();
547       
548     }
549     
550     // start the hovering countdown
551     // if the mouse does not move for 2 secs, fire the HoverEvent to initiate
552     // picking etc.
553     this.hoverTrigger = setTimeout(function() {
554 
555       this.hoverEnd_();
556       
557       var e = new X.event.HoverEvent();
558       e._x = currentMousePosition.x;
559       e._y = currentMousePosition.y;
560       
561       this.dispatchEvent(e);
562       
563       // reset the trigger
564       this.hoverTrigger = null;
565       
566     }.bind(this), 300);
567     
568   }
569   
570   // threshold the distance to avoid 'irregular' movement
571   if (Math.abs(distance.x) < 2) {
572     
573     distance.x = 0;
574     
575   }
576   if (Math.abs(distance.y) < 2) {
577     
578     distance.y = 0;
579     
580   }
581   
582   // jump out if the distance is 0 to avoid unnecessary events
583   if (distance.magnitude() == 0) {
584     
585     return;
586     
587   }
588   
589 
590   //
591   // check which mouse buttons or keys are pressed
592   //
593   if (this._leftButtonDown && !shiftDown) {
594     //
595     // LEFT MOUSE BUTTON DOWN AND NOT SHIFT DOWN
596     //
597     
598     // create a new rotate event
599     var e = new X.event.RotateEvent();
600     
601     // attach the distance vector
602     e._distance = distance;
603     
604     // attach the angle in degrees
605     e._angle = 0;
606     
607     // .. fire the event
608     this.dispatchEvent(e);
609     
610 
611   } else if (this._middleButtonDown || (this._leftButtonDown && shiftDown)) {
612     //
613     // MIDDLE MOUSE BUTTON DOWN or LEFT MOUSE BUTTON AND SHIFT DOWN
614     //
615     
616     // create a new pan event
617     var e = new X.event.PanEvent();
618     
619     // panning in general moves pretty fast, so we threshold the distance
620     // additionally
621     if (distance.x > 5) {
622       
623       distance.x = 5;
624       
625     } else if (distance.x < -5) {
626       
627       distance.x = -5;
628       
629     }
630     if (distance.y > 5) {
631       
632       distance.y = 5;
633       
634     } else if (distance.y < -5) {
635       
636       distance.y = -5;
637       
638     }
639     
640     // attach the distance vector
641     e._distance = distance;
642     
643     // .. fire the event
644     this.dispatchEvent(e);
645     
646 
647   } else if (this._rightButtonDown) {
648     //
649     // RIGHT MOUSE BUTTON DOWN
650     //
651     
652     // create a new zoom event
653     var e = new X.event.ZoomEvent();
654     
655     // set the zoom direction
656     // true if zooming in, false if zooming out
657     e._in = (distance.y > 0);
658     
659     // with the right click, the zoom will happen rather
660     // fine than fast
661     e._fast = false;
662     
663     // .. fire the event
664     this.dispatchEvent(e);
665     
666 
667   }
668   
669 };

674  */

676 
677   if (this.hoverTrigger) {
678     clearTimeout(this.hoverTrigger);
679   }
680   
681   var e = new X.event.HoverEndEvent();
682   this.dispatchEvent(e);
683   
684 };

732 /**
733  * Callback for keyboard events on the associated DOM element. This fires proper
734  * X.event events.
735  * 
736  * @param {Event} event The browser fired event.
737  * @protected
738  */
739 X.interactor.prototype.onKey_ = function(event) {
740 
741   // only listen to key events if the mouse is inside our element
742   // this f.e. enables key event listening for multiple renderers
743   if (!this._mouseInside) {
744     
745     return;
746     
747   }
748   
749   this['keyEvent'] = event; // buffering..
750   eval("this.onKey(this['keyEvent'])");
751   
752   // end all hovering since the scene can change and a caption might be
753   // misplaced etc.
754   this.hoverEnd_();
755   
756   // observe the control keys (shift, alt, ..)
757   var alt = event.altKey;
758   var ctrl = event.ctrlKey;
759   var meta = event.metaKey; // this is f.e. the windows or apple key
760   var shift = event.shiftKey;
761   
762   // get the keyCode
763   var keyCode = event.keyCode;
764   
765   if (keyCode == 82 && !alt && !ctrl && !meta && !shift) {
766     
767     // 'r' but without any other control keys since we do not want to limit the
768     // user to press for example CTRL+R to reload the page
769     
770     // prevent any other actions..
771     event.preventDefault();
772     
773     // fire the ResetViewEvent
774     var e = new X.event.ResetViewEvent();
775     this.dispatchEvent(e);
776     
777   } else if (keyCode >= 37 && keyCode <= 40) {
778     
779     // keyCode <= 37 and >= 40 means the arrow keys
780     
781     // prevent any other actions..
782     event.preventDefault();
783     
784     var e = null;
785     
786     if (shift) {
787       
788       // create a new pan event
789       e = new X.event.PanEvent();
790       
791     } else if (alt) {
792       
793       // create a new zoom event
794       e = new X.event.ZoomEvent();
795       
796     } else {
797       // create a new rotate event for 3D or a new scroll event for 2D
798       e = new X.event.RotateEvent();
799       if (this instanceof X.interactor2D) {
800         e = new X.event.ScrollEvent();
801       }
802       
803     }
804     
805     if (!e) {
806       
807       // should not happen but you never know with key interaction
808       return;
809       
810     }
811     
812     // create a distance vector
813     var distance = new goog.math.Vec2(0, 0);
814     
815     if (keyCode == 37) {
816       // '<-' LEFT
817       distance.x = 5;
818       e._up = false; // scroll direction
819       if (alt) {
820         // for zoom, we configure the zooming behavior
821         e._up = true;
822         e._in = true;
823         e._fast = false;
824       }
825       
826     } else if (keyCode == 39) {
827       // '->' RIGHT
828       distance.x = -5;
829       e._up = true; // scroll direction
830       if (alt) {
831         // for zoom, we configure the zooming behavior
832         e._in = false;
833         e._fast = false;
834       }
835       
836     } else if (keyCode == 38) {
837       // '^-' TOP
838       distance.y = 5;
839       e._up = true; // scroll direction
840       if (alt) {
841         // for zoom, we configure the zooming behavior
842         e._in = true;
843         e._fast = true;
844       }
845       
846     } else if (keyCode == 40) {
847       // '-v' BOTTOM
848       distance.y = -5;
849       e._up = false; // scroll direction
850       if (alt) {
851         // for zoom, we configure the zooming behavior
852         e._in = false;
853         e._fast = true;
854       }
855       
856     }
857     
858     // attach the distance vector
859     e._distance = distance;
860
