 44 /**
 45  * Create a camera.
 46  * 
 47  * @constructor
 48  * @param {number} width The width of the camera's viewport.
 49  * @param {number} height The height of the camera's viewport.
 50  * @extends X.base
 51  */
 52 X.camera = function(width, height) {
 53 
 54   // validate width and height
 55   if (!goog.isNumber(width) || !goog.isNumber(height)) {
 56     
 57     throw new Error('A camera needs valid width and height values.');
 58     
 59   }
 60   
 61   //
 62   // call the standard constructor of X.base
 63   goog.base(this);
 64   
 65   //
 66   // class attributes
 67   
 68   /**
 69    * @inheritDoc
 70    * @const
 71    */
 72   this._classname = 'camera';
 73   
 74   /**
 75    * The position of this camera, by default 0, 0, 100.
 76    * 
 77    * @type {!goog.math.Vec3}
 78    * @protected
 79    */
 80   this._position = new goog.math.Vec3(0, 0, 100);
 81   
 82   /**
 83    * The focus point of this camera, by default 0, 0, 0.
 84    * 
 85    * @type {!goog.math.Vec3}
 86    * @protected
 87    */
 88   this._focus = new goog.math.Vec3(0, 0, 0);
 89   
 90   /**
 91    * The unit vector pointing to the top of the three-dimensional space.
 92    * 
 93    * @type {!goog.math.Vec3}
 94    * @protected
 95    */
 96   this._up = new goog.math.Vec3(0, 1, 0);
 97   
 98   /**
 99    * The view matrix.
100    * 
101    * @type {!X.matrix}
102    * @protected
103    */
104   this._view = this.lookAt_(this._position, this._focus);
105   
106   /**
107    * The view matrix as a 'ready-to-use'-gl version.
108    * 
109    * @type {!Object}
110    * @protected
111    */
112   this._glview = new Float32Array(this._view.flatten());
113   
114 };
115 // inherit from X.base
116 goog.inherits(X.camera, X.base);

119 /**
120  * Configures observers for a given interactor. The method sets up listeners for
121  * PAN and ZOOM events.
122  * 
123  * @param {!X.interactor} interactor The interactor which should be observed.
124  */
125 X.camera.prototype.observe = function(interactor) {
126 
127   if (!goog.isDefAndNotNull(interactor) ||
128       !(interactor instanceof X.interactor)) {
129     
130     throw new Error('Could not observe the interactor.');
131     
132   }
133   
134   goog.events.listen(interactor, X.event.events.PAN, this.onPan_.bind(this));
135   goog.events.listen(interactor, X.event.events.ZOOM, this.onZoom_.bind(this));
136   
137 };

140 /**
141  * The callback for a ZOOM event.
142  * 
143  * @param {!X.event.ZoomEvent} event The event.
144  * @throws {Error} An exception if the event is invalid.
145  * @protected
146  */
147 X.camera.prototype.onZoom_ = function(event) {
148 
149   if (!(event instanceof X.event.ZoomEvent)) {
150     
151     throw new Error('Received no valid zoom event.');
152     
153   }
154   
155   if (event._in) {
156     
157     this.zoomIn(event._fast);
158     
159   } else {
160     
161     this.zoomOut(event._fast);
162     
163   }
164   
165 };

170  * 
171  * @param {!X.event.PanEvent} event The event.
172  * @throws {Error} An exception if the event is invalid.
173  * @protected
174  */
175 X.camera.prototype.onPan_ = function(event) {
176 
177   if (!(event instanceof X.event.PanEvent)) {
178     
179     throw new Error('Received no valid pan event.');
180     
181   }
182   
183   this.pan(event._distance);
184   
185 };

233 /**
234  * Set the position of this camera. This forces a re-calculation of the view
235  * matrix. This action _does not_ force an immediate render event automatically.
236  * 
237  * @param {!Array} position The X,Y,Z components of the position as an array.
238  * @throws {Error} An error, if the position was invalid.
239  * @public
240  */
241 X.camera.prototype.__defineSetter__('position', function(position) {
242 
243   if (!goog.isDefAndNotNull(position) || !(position instanceof Array) ||
244       (position.length != 3)) {
245     
246     throw new Error('Invalid position.');
247     
248   }
249   
250   var x = position[0];
251   var y = position[1];
252   var z = position[2];
253   
254   this._position = new goog.math.Vec3(x, y, z);
255   
256   // we need to reset to re-calculate the matrices
257   this.reset();
258   
259 });

344 /**
345  * Reset the camera according to its configured position and focus.
346  */
347 X.camera.prototype.reset = function() {
348 
349   // update the view matrix and its gl versions
350   this._view = this.lookAt_(this._position, this._focus);
351   this._glview = new Float32Array(this._view.flatten());
352   
353 };

356 /**
357  * Perform a pan operation. This method fires a X.event.RenderEvent() after the
358  * calculation is done.
359  * 
360  * @param {!goog.math.Vec2} distance The distance of the panning in respect of
361  *          the last camera position.
362  * @public
363  */
364 X.camera.prototype.pan = function(distance) {
365 
366   if (!(distance instanceof goog.math.Vec2)) {
367     
368     throw new Error('Invalid distance vector for pan operation.');
369     
370   }
371   
372   var distance3d = new goog.math.Vec3(-distance.x, distance.y, 0);
373   
374   var identity = X.matrix.createIdentityMatrix(4);
375   var panMatrix = identity.translate(distance3d);
376   
377   this._view = new X.matrix(panMatrix.multiply(this._view));
378   this._glview = new Float32Array(this._view.flatten());
379   
380   // fire a render event
381   this.dispatchEvent(new X.event.RenderEvent());
382   
383 };

386 /**
387  * Perform a zoom in operation. This method fires a X.event.RenderEvent() after
388  * the calculation is done.
389  * 
390  * @param {boolean} fast Enables/disables the fast mode which zooms much
391  *          quicker.
392  * @public
393  */
394 X.camera.prototype.zoomIn = function(fast) {
395 
396   var zoomStep = 20;
397   
398   if (goog.isDefAndNotNull(fast) && !fast) {
399     
400     zoomStep = 1;
401     
402   }
403   
404   var zoomVector = new goog.math.Vec3(0, 0, zoomStep);
405   
406   var identity = X.matrix.createIdentityMatrix(4);
407   var zoomMatrix = identity.translate(zoomVector);
408   
409   this._view = new X.matrix(zoomMatrix.multiply(this._view));
410   this._glview = new Float32Array(this._view.flatten());
411   
412   // fire a render event
413   this.dispatchEvent(new X.event.RenderEvent());
414   
415 };
416

418 /**
419  * Perform a zoom out operation. This method fires a X.event.RenderEvent() after
420  * the calculation is done.
421  * 
422  * @param {boolean} fast Enables/disables the fast mode which zooms much
423  *          quicker.
424  * @public
425  */
426 X.camera.prototype.zoomOut = function(fast) {
427 
428   var zoomStep = 30;
429   
430   if (goog.isDefAndNotNull(fast) && !fast) {
431     
432     zoomStep = 1;
433     
434   }
435   
436   var zoomVector = new goog.math.Vec3(0, 0, -zoomStep);
437   
438   var identity = X.matrix.createIdentityMatrix(4);
439   var zoomMatrix = identity.translate(zoomVector);
440   
441   this._view = new X.matrix(zoomMatrix.multiply(this._view));
442   this._glview = new Float32Array(this._view.flatten());
443   
444   // fire a render event
445   this.dispatchEvent(new X.event.RenderEvent());
446   
447 };

450 /**
451  * Calculate a view matrix by using the camera position and a focus point.
452  * 
453  * @param {!goog.math.Vec3} cameraPosition The camera position.
454  * @param {!goog.math.Vec3} targetPoint The focus (target) point.
455  * @return {!X.matrix} The view matrix.
456  * @throws {Error} If the given arguments are invalid.
457  * @protected
458  */
459 X.camera.prototype.lookAt_ = function(cameraPosition, targetPoint) {
460 
461   if (!(cameraPosition instanceof goog.math.Vec3) ||
462       !(targetPoint instanceof goog.math.Vec3)) {
463     
464     throw new Error('3D vectors required for calculating the view.');
465     
466   }
467   
468   return X.matrix.createIdentityMatrix(4);
469   
470 };
