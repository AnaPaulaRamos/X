 39 /**
 40  * Create a matrix as an object based on goog.math.Matrix but with additional
 41  * functionality.
 42  * 
 43  * @constructor
 44  * @param {goog.math.Matrix|Array.<Array.<number>>|goog.math.Size|number} m A
 45  *          matrix to copy, a 2D-array to take as a template, a size object for
 46  *          dimensions, or the number of rows.
 47  * @param {number=} opt_n Number of columns of the matrix (only applicable if
 48  *          the first argument is also numeric).
 49  * @extends goog.math.Matrix
 50  */
 51 X.matrix = function(m, opt_n) {
 52 
 53   //
 54   // call the standard constructor
 55   goog.base(this, m, opt_n);
 56   
 57   //
 58   // class attributes
 59   
 60   /**
 61    * The className of this class.
 62    * 
 63    * @type {string}
 64    * @protected
 65    */
 66   this._classname = 'matrix';
 67   
 68 };
 69 // inherit from goog.math.Matrix
 70 goog.inherits(X.matrix, goog.math.Matrix);

 73 /**
 74  * Creates a square identity matrix. i.e. for n = 3:
 75  * 
 76  * <pre>
 77  * [ 1 0 0 ]
 78  * [ 0 1 0 ]
 79  * [ 0 0 1 ]
 80  * </pre>
 81  * 
 82  * @param {number} n The size of the square identity matrix.
 83  * @return {!X.matrix} Identity matrix of width and height {@code n}.
 84  */
 85 X.matrix.createIdentityMatrix = function(n) {
 86 
 87   // (c) Google
 88   // Code taken from goog.math.Matrix since the 'static' methods are not
 89   // inherited..
 90   var rv = [];
 91   for ( var i = 0; i < n; i++) {
 92     rv[i] = [];
 93     for ( var j = 0; j < n; j++) {
 94       rv[i][j] = i == j ? 1 : 0;
 95     }
 96   }
 97   return new X.matrix(rv);
 98 };

101 /**
102  * Create a flattened, one-dimensional representation of a matrix. Notice: This
103  * method flattens column by column.
104  * 
105  * @this {X.matrix}
106  * @return {Array} A one-dimensional representation of this matrix.
107  */
108 X.matrix.prototype.flatten = function() {
109 
110   var result = [];
111   
112   var dimensions = this.getSize();
113   
114   if (dimensions.height == 0 || dimensions.width == 0) {
115     return [];
116   }
117   
118   var i, j;
119   for (j = 0; j < dimensions.height; j++) {
120     for (i = 0; i < dimensions.width; i++) {
121       result.push(this.getValueAt(i, j));
122     }
123   }
124   return result;
125   
126 };

129 /**
130  * Translate a 3x3 or 4x4 matrix by a vector. In the 3x3 case, the vector has to
131  * be 2 dimensional. In the 4x4 case, the vector has to be 3 dimensional.
132  * 
133  * @this {X.matrix}
134  * @param {!goog.math.Vec2|!goog.math.Vec3} vector The translation vector.
135  * @return {!X.matrix} The result of this translation.
136  * @throws {Error} An exception if the translation fails.
137  */
138 X.matrix.prototype.translate = function(vector) {
139 
140   if (!this.isSquare()) {
141     
142     throw new Error('Can not translate non-square matrix.');
143     
144   }
145   
146   var dimensions = this.getSize();
147   
148   var transformationMatrix = goog.math.Matrix
149       .createIdentityMatrix(dimensions.height);
150   
151   if (vector instanceof goog.math.Vec2 && dimensions.height == 3) {
152     
153     transformationMatrix.setValueAt(0, 2, vector.x);
154     transformationMatrix.setValueAt(1, 2, vector.y);
155     
156   } else if (vector instanceof goog.math.Vec3 && dimensions.height == 4) {
157     
158     transformationMatrix.setValueAt(0, 3, vector.x);
159     transformationMatrix.setValueAt(1, 3, vector.y);
160     transformationMatrix.setValueAt(2, 3, vector.z);
161     
162   } else {
163     
164     throw new Error('Translation failed.');
165     
166   }
167   
168   // now multiply this matrix with the transformationMatrix and return it..
169   return new X.matrix(this.multiply(transformationMatrix));
170   
171 };

174 /**
175  * Rotate a 4x4 matrix by an angle around an axis.
176  * 
177  * @this {X.matrix}
178  * @param {!number} angle The rotation angle.
179  * @param {!goog.math.Vec3} iaxis The axis to rotate around.
180  * @return {!X.matrix} The result of this rotation.
181  * @throws {Error} An exception if the rotation fails.
182  */
183 X.matrix.prototype.rotate = function(angle, iaxis) {
184 
185   var dimensions = this.getSize();
186   
187   if (dimensions.height != 4 || !this.isSquare()) {
188     
189     throw new Error('Only 4x4 matrices can be rotated.');
190     
191   }
192   
193   if (!goog.isDefAndNotNull(iaxis) || !(iaxis instanceof goog.math.Vec3)) {
194     
195     throw new Error('Invalid axis vector.');
196     
197   }
198   
199   if (!goog.isNumber(angle)) {
200     
201     throw new Error('Invalid angle.');
202     
203   }
204   
205   // normalize the axis
206   var axis = iaxis.normalize();
207   
208   // trigonometrical fun
209   var cos = Math.cos(angle);
210   var sin = Math.sin(angle);
211   
212   var rotationMatrix = goog.math.Matrix.createIdentityMatrix(4);
213   
214   // build rotation matrix according to
215   // http://en.wikipedia.org/wiki/Rotation_matrix#Rotation_matrix_from_axis_and_angle
216   
217   rotationMatrix.setValueAt(0, 0, cos + axis.x * axis.x * (1 - cos));
218   rotationMatrix.setValueAt(0, 1, axis.x * axis.y * (1 - cos) - axis.z * sin);
219   rotationMatrix.setValueAt(0, 2, axis.x * axis.z * (1 - cos) + axis.y * sin);
220   
221   rotationMatrix.setValueAt(1, 0, axis.y * axis.x * (1 - cos) + axis.z * sin);
222   rotationMatrix.setValueAt(1, 1, cos + axis.y * axis.y * (1 - cos));
223   rotationMatrix.setValueAt(1, 2, axis.y * axis.z * (1 - cos) - axis.x * sin);
224   
225   rotationMatrix.setValueAt(2, 0, axis.z * axis.x * (1 - cos) - axis.y * sin);
226   rotationMatrix.setValueAt(2, 1, axis.z * axis.y * (1 - cos) + axis.x * sin);
227   rotationMatrix.setValueAt(2, 2, cos + axis.z * axis.z * (1 - cos));
228   
229   // now multiply and return
230   return new X.matrix(this.multiply(rotationMatrix));
231   
232 };

235 /**
236  * Multiply 3x3 or 4x4 matrix by a vector. The vector has to be at least 3
237  * dimensional.
238  * 
239  * @this {X.matrix}
240  * @param {!goog.math.Vec3} vector The multiplication vector.
241  * @return {!goog.math.Vec3} The result of this multiplication.
242  * @throws {Error} An exception if the multiplication fails.
243  */
244 X.matrix.prototype.multiplyByVector = function(vector) {
245 
246   var dimensions = this.getSize();
247   
248   // we need to convert the vector to a matrix
249   //
250   var vectorAsArray = new Array(dimensions.width);
251   
252   var i;
253   for (i = 0; i < vectorAsArray.length; i++) {
254     
255     vectorAsArray[i] = new Array(1);
256     vectorAsArray[i][0] = 1;
257     
258   }
259   
260   if (vector instanceof goog.math.Vec3 && dimensions.width >= 3) {
261     
262     vectorAsArray[0][0] = vector.x;
263     vectorAsArray[1][0] = vector.y;
264     vectorAsArray[2][0] = vector.z;
265     
266   } else {
267     
268     throw new Error('Multiplication by vector failed.');
269     
270   }
271   
272   // now convert the vectorAsArray to a matrix and multiply
273   var vectorAsMatrix = new goog.math.Matrix(vectorAsArray);
274   
275   // ...and multiply it
276   var resultMatrix = this.multiply(vectorAsMatrix);
277   
278   return new goog.math.Vec3(parseFloat(resultMatrix.getValueAt(0, 0)),
279       parseFloat(resultMatrix.getValueAt(1, 0)), parseFloat(resultMatrix
280           .getValueAt(2, 0)));
281   
282 };

298 /**
299  * Swap two columns.
300  * 
301  * @param {number} col1 Index of the first column.
302  * @param {number} col2 Index of the second column.
303  */
304 X.matrix.prototype.swapCols = function(col1, col2) {
305 
306   var dimensions = this.getSize();
307   
308   var h = 0;
309   for (h = 0; h < dimensions.height; h++) {
310     
311     var tmpValue = this.getValueAt(h, col1);
312     if (!tmpValue) {
313       tmpValue = 0;
314     }
315     
316     var tmpValue2 = this.getValueAt(h, col2);
317     if (!tmpValue2) {
318       tmpValue2 = 0;
319     }
320     
321     this.setValueAt(h, col1, tmpValue2);
322     this.setValueAt(h, col2, tmpValue);
323     
324   }
325   
326 };
