 30 // provides
 31 goog.provide('X.labelmap');
 32 
 33 // requires
 34 goog.require('X.volume');
 35 
 36 
 37 
 38 /**
 39  * Pseudo-class for a X.labelmap which derives from X.volume and is used to
 40  * distinguish between a volume and a label map. An X.labelmap will never be
 41  * rendered separately - but an X.volume object can be used to display solely a
 42  * label map.
 43  * 
 44  * @constructor
 45  * @extends X.volume
 46  */
 47 X.labelmap = function(volume) {
 48 
 49   //
 50   // call the standard constructor of X.volume
 51   goog.base(this);
 52   
 53   //
 54   // class attributes
 55   
 56   /**
 57    * @inheritDoc
 58    * @const
 59    */
 60   this._classname = 'labelmap';
 61   
 62   this._volume = volume;
 63   
 64 };
 65 // inherit from X.volume
 66 goog.inherits(X.labelmap, X.volume);
 67 
 68 
 69 /**
 70  * Re-show the slices or re-activate the volume rendering for this volume.
 71  * 
 72  * @inheritDoc
 73  */
 74 X.labelmap.prototype.modified = function() {
 75 
 76   // .. and fire our own modified event
 77   var modifiedEvent = new X.event.ModifiedEvent();
 78   modifiedEvent._object = this;
 79   this.dispatchEvent(modifiedEvent);
 80   
 81   // call the X.volumes' modified method
 82   this._volume.modified();
 83   
 84 };
 85 
 86 
 87 // export symbols (required for advanced compilation and in particular the copy
 88 // constructors with duck typing)
 89 goog.exportSymbol('X.labelmap', X.labelmap);
