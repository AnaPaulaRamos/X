 30 goog.provide('X.interactor2D');
 31 
 32 // requires
 33 goog.require('X.interactor');
 34 goog.require('X.event.ScrollEvent');
 35 
 36 
 37 
 38 /**
 39  * Create a 2D interactor for a given element in the DOM tree.
 40  * 
 41  * @constructor
 42  * @param {Element} element The DOM element to be observed.
 43  * @extends X.interactor
 44  */
 45 X.interactor2D = function(element) {
 46 
 47   //
 48   // call the standard constructor of X.base
 49   goog.base(this, element);
 50   
 51   //
 52   // class attributes
 53   
 54   /**
 55    * @inheritDoc
 56    * @const
 57    */
 58   this._classname = 'interactor2D';
 59   
 60 };
 61 // inherit from X.base
 62 goog.inherits(X.interactor2D, X.interactor);
 63 
 64 
 65 /**
 66  * @inheritDoc
 67  */
 68 X.interactor2D.prototype.onMouseWheel_ = function(event) {
 69 
 70   goog.base(this, 'onMouseWheel_', event);
 71   
 72   // create a new scroll event
 73   //
 74   // the scroll event triggers scrolling through the slices in a 2D renderer. we
 75   // can not use the zoom event here since we also zoom on right click.
 76   var e = new X.event.ScrollEvent();
 77   
 78   // make sure, deltaY is defined
 79   if (!goog.isDefAndNotNull(event.deltaY)) {
 80     event.deltaY = 0;
 81   }
 82   
 83   // set the scroll direction
 84   // true if up, false if down
 85   // delta is here given by the event
 86   e._up = (event.deltaY < 0);
 87   
 88   // .. fire the event
 89   this.dispatchEvent(e);
 90   
 91 };
