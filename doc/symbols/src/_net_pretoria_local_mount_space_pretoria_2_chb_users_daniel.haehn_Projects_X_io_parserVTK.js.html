 41 /**
 42  * Create a parser for the ascii .VTK format.
 43  * 
 44  * @constructor
 45  * @extends X.parser
 46  */
 47 X.parserVTK = function() {
 48 
 49   //
 50   // call the standard constructor of X.base
 51   goog.base(this);
 52   
 53   //
 54   // class attributes
 55   
 56   /**
 57    * @inheritDoc
 58    * @const
 59    */
 60   this._classname = 'parserVTK';
 61   
 62 };
 63 // inherit from X.parser
 64 goog.inherits(X.parserVTK, X.parser);

 67 /**
 68  * @inheritDoc
 69  */
 70 X.parserVTK.prototype.parse = function(container, object, data, flag) {
 71 
 72   var p = object._points;
 73   var n = object._normals;
 74   
 75   var dataAsArray = data.split('\n');
 76   var numberOfLines = dataAsArray.length;
 77   
 78   // in .VTK files, the points are not ordered for rendering, so we need to
 79   // buffer everything in X.triplets containers and then order it
 80   var unorderedPoints = new X.triplets();
 81   var unorderedNormals = new X.triplets();
 82   
 83   // .. we also need a buffer for all indices
 84   this._geometries = [];
 85   
 86   // even if vtk files support multiple object types in the same file, we only
 87   // support one kind
 88   this._objectType = X.displayable.types.TRIANGLES;
 89   
 90   // this mode indicates that the next lines will be X,Y,Z coordinates
 91   this._pointsMode = false;
 92   
 93   // this mode indicates that the next lines will be indices mapping to points
 94   // and pointData
 95   this._geometryMode = false;
 96   
 97   // this mode indicates that the next lines will be pointData
 98   this._pointDataMode = false;
 99   // one type of pointData are normals and right now the only supported ones
100   this._normalsMode = false;
101   
102 
103   //
104   // LOOP THROUGH ALL LINES
105   //
106   // This uses an optimized loop.
107   //
108   
109   /*
110    * Fast Duff's Device
111    * 
112    * @author Miller Medeiros <http://millermedeiros.com>
113    * 
114    * @version 0.3 (2010/08/25)
115    */
116   var i = 0;
117   var n2 = numberOfLines % 8;
118   while (n2--) {
119     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
120     i++;
121   }
122   
123   n2 = (numberOfLines * 0.125) ^ 0;
124   while (n2--) {
125     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
126     i++;
127     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
128     i++;
129     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
130     i++;
131     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
132     i++;
133     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
134     i++;
135     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
136     i++;
137     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
138     i++;
139     this.parseLine(unorderedPoints, unorderedNormals, dataAsArray[i]);
140     i++;
141   }
142   
143 
144   // now, configure the object according to the objectType
145   this.configure(unorderedPoints, unorderedNormals, p, n);
146   
147   // .. and set the objectType
148   object._type = this._objectType;
149   
150   // the object should be set up here, so let's fire a modified event
151   var modifiedEvent = new X.event.ModifiedEvent();
152   modifiedEvent._object = object;
153   modifiedEvent._container = container;
154   this.dispatchEvent(modifiedEvent);
155   
156 };

159 /**
160  * Parses a line of .VTK data and modifies the given X.triplets containers.
161  * 
162  * @param {!X.triplets} unorderedPoints A points container as a X.triplets
163  *          object.
164  * @param {!X.triplets} unorderedNormals A normals container as a X.triplets
165  *          object.
166  * @param {!string} line The line to parse.
167  * @protected
168  */
169 X.parserVTK.prototype.parseLine = function(unorderedPoints, unorderedNormals,
170     line) {
171 
172   // trim the line
173   line = line.replace(/^\s+|\s+$/g, '');
174   
175   // split to array
176   var lineFields = line.split(' ');
177   
178   // number of lineFields
179   var numberOfLineFields = lineFields.length;
180   
181   // the first field of the line can be a keyword to indicate different modes
182   var firstLineField = lineFields[0];
183   
184   // KEYWORD CHECK / MODE SWITCH
185   //
186   // identify the section of the next coming lines using the vtk keywords
187   switch (firstLineField) {
188   
189   case 'POINTS':
190 
191     // this means that real X,Y,Z points are coming
192     
193     this._pointsMode = true;
194     this._geometryMode = false;
195     this._pointDataMode = false;
196     
197     // go to next line
198     return;
199     
200   case 'VERTICES':
201 
202     // this means that triangles or points are coming
203     
204     this._geometryMode = true;
205     this._pointsMode = false;
206     this._pointDataMode = false;
207     
208     var numberOfElements = parseInt(lineFields[1], 10);
209     
210     if (numberOfElements >= 3) {
211       this._objectType = X.displayable.types.TRIANGLES;
212     } else if (numberOfElements == 1) {
213       this._objectType = X.displayable.types.POINTS;
214     } else {
215       
216       throw new Error('This VTK file is not supported!');
217       
218     }
219     
220     // reset all former geometries since we only support 1 geometry type per
221     // file (the last one specified)
222     this._geometries = [];
223     
224     // go to next line
225     return;
226     
227   case 'TRIANGLE_STRIPS':
228 
229     // this means that triangle_strips are coming
230     
231     this._geometryMode = true;
232     this._pointsMode = false;
233     this._pointDataMode = false;
234     this._objectType = X.displayable.types.TRIANGLE_STRIPS;
235     
236     // reset all former geometries since we only support 1 geometry type per
237     // file (the last one specified)
238     this._geometries = [];
239     
240     // go to next line
241     return;
242     
243   case 'LINES':
244 
245     // this means that lines are coming
246     
247     this._geometryMode = true;
248     this._pointsMode = false;
249     this._pointDataMode = false;
250     this._objectType = X.displayable.types.LINES;
251     
252     // reset all former geometries since we only support 1 geometry type per
253     // file (the last one specified)
254     this._geometries = [];
255     
256     // go to next line
257     return;
258     
259   case 'POLYGONS':
260 
261     // this means that polygons are coming
262     // we only support polygons which are triangles right now
263     
264     this._geometryMode = true;
265     this._pointsMode = false;
266     this._pointDataMode = false;
267     this._objectType = X.displayable.types.POLYGONS;
268     
269     // reset all former geometries since we only support 1 geometry type per
270     // file (the last one specified)
271     this._geometries = [];
272     
273     // go to next line
274     return;
275     
276   case 'POINT_DATA':
277 
278     // this means point-data is coming
279     // f.e. normals
280     
281     this._pointDataMode = true;
282     this._pointsMode = false;
283     this._geometryMode = false;
284     
285     // go to next line
286     return;
287     
288   }
289   
290   // PARSING
291   //
292   // now we parse according to the current mode
293   //
294   if (this._pointsMode) {
295     
296     // in pointsMode, each line has X,Y,Z coordinates separated by space
297     
298     if (numberOfLineFields == 1 || isNaN(parseFloat(firstLineField))) {
299       
300       // this likely means end of pointsMode
301       this._pointsMode = false;
302       
303       return;
304       
305     }
306     
307     // assume max. 9 coordinate values (== 3 points) in one row
308     if (numberOfLineFields >= 3) {
309       var x0 = parseFloat(lineFields[0]);
310       var y0 = parseFloat(lineFields[1]);
311       var z0 = parseFloat(lineFields[2]);
312       
313       unorderedPoints.add(x0, y0, z0);
314     }
315     
316     if (numberOfLineFields >= 6) {
317       var x1 = parseFloat(lineFields[3]);
318       var y1 = parseFloat(lineFields[4]);
319       var z1 = parseFloat(lineFields[5]);
320       unorderedPoints.add(x1, y1, z1);
321     }
322     
323     if (numberOfLineFields >= 9) {
324       var x2 = parseFloat(lineFields[6]);
325       var y2 = parseFloat(lineFields[7]);
326       var z2 = parseFloat(lineFields[8]);
327       
328       unorderedPoints.add(x2, y2, z2);
329     }
330     
331   } // end of pointsMode
332   else if (this._geometryMode) {
333     
334     // in geometryMode, each line has indices which map to points and pointsData
335     
336     if (numberOfLineFields == 1 || isNaN(parseFloat(firstLineField))) {
337       
338       // this likely means end of geometryMode
339       this._geometryMode = false;
340       return;
341       
342     }
343     
344     // the first element is the number of coming indices
345     // so we just slice the first element to get all indices
346     var values = lineFields.slice(1);
347     
348     // append all index values to the main geometries array
349     this._geometries.push(values);
350     
351   } // end of geometryMode
352   else if (this._pointDataMode) {
353     
354     // at the moment, only normals are supported as point-data
355     
356     if (firstLineField == 'NORMALS') {
357       
358       this._normalsMode = true;
359       
360       return;
361       
362     }
363     
364     if (numberOfLineFields == 1 || isNaN(parseFloat(firstLineField))) {
365       
366       // this likely means end of pointDataMode
367       this._pointDataMode = false;
368       this._normalsMode = false;
369       
370       return;
371       
372     }
373     
374     // the normals mode
375     if (this._normalsMode) {
376       
377       // assume 9 coordinate values (== 3 points) in one row
378       
379       if (numberOfLineFields >= 3) {
380         var x0 = parseFloat(lineFields[0]);
381         var y0 = parseFloat(lineFields[1]);
382         var z0 = parseFloat(lineFields[2]);
383         unorderedNormals.add(x0, y0, z0);
384       }
385       if (numberOfLineFields >= 6) {
386         var x1 = parseFloat(lineFields[3]);
387         var y1 = parseFloat(lineFields[4]);
388         var z1 = parseFloat(lineFields[5]);
389         unorderedNormals.add(x1, y1, z1);
390       }
391       if (numberOfLineFields >= 9) {
392         var x2 = parseFloat(lineFields[6]);
393         var y2 = parseFloat(lineFields[7]);
394         var z2 = parseFloat(lineFields[8]);
395         unorderedNormals.add(x2, y2, z2);
396       }
397       
398     } // end of normalsMode
399     
400   } // end of pointDataMode
401   
402 };

405 /**
406  * Configure X.object points and normals. This method takes the object type into
407  * consideration to f.e. use degenerated triangles for TRIANGLE_STRIPS.
408  * 
409  * @param {!X.triplets} unorderedPoints The container for unordered points.
410  * @param {!X.triplets} unorderedNormals The container for unordered normals.
411  * @param {!X.triplets} p The points container of the X.object.
412  * @param {!X.triplets} n The normals container of the X.object.
413  */
414 X.parserVTK.prototype.configure = function(unorderedPoints, unorderedNormals,
415     p, n) {
416 
417   // cache often used values for fast access
418   var numberOfUnorderedNormals = unorderedNormals.length;
419   
420   var numberOfGeometries = this._geometries.length;
421   var i = numberOfGeometries;
422   // we use this loop here since it's slightly faster than the for loop
423   do {
424     
425     // we want to loop through the geometries in the range 0..(N - 1)
426     var currentGeometry = this._geometries[numberOfGeometries - i];
427     var currentGeometryLength = currentGeometry.length;
428     
429     // in the sub-loop we loop through the indices of the current geometry
430     var k;
431     for (k = 0; k < currentGeometryLength; k++) {
432       
433       // boundary check for LINES
434       if (this._objectType == X.displayable.types.LINES &&
435           (k + 1 >= currentGeometryLength)) {
436         
437         // jump out since we reached the end of the geometry
438         break;
439         
440       }
441       
442       // grab the current index
443       var currentIndex = parseInt(currentGeometry[k], 10);
444       
445       // grab the point with the currentIndex
446       var currentPoint = unorderedPoints.get(currentIndex);
447       
448       //
449       // POINTS
450       //
451       
452       // .. and add it
453       p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
454       
455       var nextIndex = currentIndex;
456       var nextPoint = currentPoint;
457       // special case for LINES: we add the next element twice to
458       // interrupt the line segments (in webGL, lines mode connects always 2
459       // points)
460       // if we would not do this, then all line segments would be connected
461       if (this._objectType == X.displayable.types.LINES) {
462         
463         nextIndex = parseInt(currentGeometry[k + 1], 10);
464         // grab the next point
465         nextPoint = unorderedPoints.get(nextIndex);
466         
467         // and add it
468         p.add(nextPoint[0], nextPoint[1], nextPoint[2]);
469         
470       } // LINES
471       
472       // special case for TRIANGLE_STRIPS: we add the first and the
473       // last element twice to interrupt the strips (as degenerated triangles)
474       // if we would not do this, then all strips would be connected
475       else if (this._objectType == X.displayable.types.TRIANGLE_STRIPS) {
476         
477         // check if this is the first or last element
478         if (k == 0 || k == currentGeometryLength - 1) {
479           
480           // if this is the first or last point of the triangle strip, add it
481           // again
482           p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
483           
484         }
485         
486       } // TRIANGLE_STRIPS
487       
488 
489 
490       //
491       // NORMALS
492       // 
493       if (currentIndex < numberOfUnorderedNormals) {
494         
495         // grab the normal with the currentIndex, if it exists
496         var currentNormals = unorderedNormals.get(currentIndex);
497         
498         // .. and add it
499         n.add(currentNormals[0], currentNormals[1], currentNormals[2]);
500         
501         // for LINES, add the next normal (neighbor)
502         if (this._objectType == X.displayable.types.LINES) {
503           
504           // the neighbor
505           var nextNormals = unorderedNormals.get(nextIndex);
506           
507           // .. and add it
508           n.add(nextNormals[0], nextNormals[1], nextNormals[2]);
509           
510         } // LINES
511         
512         // for TRIANGLE_STRIPS, special case
513         else if (this._objectType == X.displayable.types.TRIANGLE_STRIPS) {
514           
515           // check if this is the first or last element
516           if (k == 0 || k == currentGeometryLength - 1) {
517             
518             // if this is the first or last point of the triangle strip, add it
519             // again
520             n.add(currentNormals[0], currentNormals[1], currentNormals[2]);
521             
522           }
523           
524         } // TRIANGLE_STRIPS
525         
526 
527       } else {
528         
529         // add an artificial normal
530         var artificialNormal = new goog.math.Vec3(currentPoint[0],
531             currentPoint[1], currentPoint[2]);
532         artificialNormal.normalize();
533         n.add(artificialNormal.x, artificialNormal.y, artificialNormal.z);
534         
535         // for LINES, do it again
536         if (this._objectType == X.displayable.types.LINES) {
537           
538           // grab the next normal (artificial)
539           var artificialNormal2 = new goog.math.Vec3(nextPoint[0],
540               nextPoint[1], nextPoint[2]);
541           artificialNormal2.normalize();
542           n.add(artificialNormal2.x, artificialNormal2.y, artificialNormal2.z);
543           
544         } // LINES
545         
546         // for TRIANGLE_STRIPS, special case
547         else if (this._objectType == X.displayable.types.TRIANGLE_STRIPS) {
548           
549           // check if this is the first or last element
550           if (k == 0 || k == currentGeometryLength - 1) {
551             
552             // add the artificial normal again
553             n.add(artificialNormal.x, artificialNormal.y, artificialNormal.z);
554             
555           }
556           
557         } // TRIANGLE_STRIPS
558         
559       }
560       
561     } // for loop through the currentGeometry
562     
563     i--;
564     
565   } while (i > 0);
566   
567 };
