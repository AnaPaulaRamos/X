 30 // provides
 31 goog.provide('X.progressbar');

 33 // requires
 34 goog.require('X.base');
 35 goog.require('goog.dom');
 36 goog.require('goog.ui.ProgressBar');

 39 /**
 40  * Create a progress bar.
 41  * 
 42  * @constructor
 43  * @param {?Element} parent The parent element in the DOM tree.
 44  * @param {!number} initialvalue An initial value for this progress bar.
 45  * @extends goog.ui.ProgressBar
 46  */
 47 X.progressbar = function(parent, initialvalue) {
 48 
 49   // check if we have a valid parent
 50   if (!goog.isDefAndNotNull(parent)) {
 51     
 52     throw new Error('No valid parent element.');
 53     
 54   }
 55   
 56   // check if we have an initial value
 57   if (!goog.isDefAndNotNull(initialvalue)) {
 58     
 59     throw new Error('Invalid initial value.');
 60     
 61   }
 62   
 63   //
 64   // call the standard constructor of goog.ui.ProgressBar
 65   goog.base(this);
 66   
 67   //
 68   // class attributes
 69   
 70   /**
 71    * The className of this class.
 72    * 
 73    * @type {string}
 74    * @protected
 75    */
 76   this._classname = 'progressbar';
 77   
 78   /**
 79    * The parent element in the DOM tree of this progress bar.
 80    * 
 81    * @type {?Element}
 82    * @protected
 83    */
 84   this._parent = parent;
 85   
 86   /**
 87    * The element for the CSS style of this progress bar.
 88    * 
 89    * @type {?Element}
 90    * @protected
 91    */
 92   this._style = null;
 93   
 94   /**
 95    * A green, full progressBar to replace this one once it's done.
 96    * 
 97    * @type {?X.progressbar}
 98    * @protected
 99    */
100   this._doneProgressBar = null;
101   
102   /**
103    * The collection of CSS definitions.
104    * 
105    * @type {!Array}
106    * @protected
107    */
108   this._css = [];
109   // configure some styles
110   var css1 = '.progress-bar-horizontal {\n';
111   css1 += '  position: relative;\n';
112   css1 += '  border: 1px solid #949dad;\n';
113   css1 += '  background: white;\n';
114   css1 += '  padding: 1px;\n';
115   css1 += '  overflow: hidden;\n';
116   css1 += '  margin: 2px;\n';
117   css1 += '  width: 100px;\n';
118   css1 += '  height: 5px;\n';
119   css1 += '}';
120   var css2 = '.progress-bar-thumb {\n';
121   css2 += '  position: relative;\n';
122   css2 += '  background: #F62217;\n';
123   css2 += '  overflow: hidden;\n';
124   css2 += '  width: 0%;\n';
125   css2 += '  height: 100%;\n';
126   css2 += '}';
127   var css3 = '.progress-bar-thumb-done {\n';
128   css3 += '  background: #57E964;\n';
129   css3 += '}';
130   this._css = [css1, css2, css3];
131   
132   // set initial value
133   this.setValue(initialvalue);
134   
135   // now initialize
136   this.init_();
137   
138 };
139 // inherit from X.base
140 goog.inherits(X.progressbar, goog.ui.ProgressBar);
141 
142 /**
143  * Internal function to initialize the progress bar which gets called by the
144  * constructor.
145  * 
146  * @private
147  */
148 X.progressbar.prototype.init_ = function() {
149 
150   // enable relative positioning for the main container
151   // this is required to place the progressBar in the center
152   
153   if (this._parent.style.position == 'static' ||
154       this._parent.style.position == '') {
155     this._parent.style.position = 'relative';
156   }
157   
158   //
159   // apply CSS styles to the document
160   //
161   var head = goog.dom.getDocument().getElementsByTagName('head')[0];
162   var style = goog.dom.createDom('style');
163   style.type = 'text/css';
164   style.media = 'screen';
165   var css = goog.dom.createTextNode(this._css[0]);
166   var css2 = goog.dom.createTextNode(this._css[1]);
167   var css3 = goog.dom.createTextNode(this._css[2]);
168   goog.dom.appendChild(head, style);
169   goog.dom.appendChild(style, css);
170   goog.dom.appendChild(style, css2);
171   goog.dom.appendChild(style, css3);
172   // save the style element for later
173   this._style = style;
174   
175   // render..
176   this.render(this._parent);
177   
178   // place the progressBar in the center
179   var pbElement = this.getElement();
180   pbElement.style.position = 'absolute';
181   pbElement.style.top = (this._parent.clientHeight - 5) / 2;
182   pbElement.style.left = (this._parent.clientWidth - 100) / 2;
183   
184 };
185

187 /**
188  * Display a green, full progress bar.
189  */
190 X.progressbar.prototype.done = function() {
191 
192   // save the position of the original progress bar
193   var top = this.getElement().style.top;
194   var left = this.getElement().style.left;
195   
196   // .. destroy the original progress bar (on the DOM side)
197   goog.dom.removeNode(this.getElement());
198   
199   // create a 'fake' new progress bar in place of the original one
200   // the new one always shows 100
201   var pb = new X.progressbar(this._parent, 100);
202   
203   // place the progressBar in the center
204   var pbElement = pb.getElement();
205   pbElement.style.position = 'absolute';
206   pbElement.style.top = top;
207   pbElement.style.left = left;
208   
209   // get the actual progress element of the progressBar
210   var pbBar = goog.dom.getFirstElementChild(pbElement);
211   
212   // .. change the color to green
213   pbBar.classList.add('progress-bar-thumb-done');
214   
215   this._doneProgressBar = pb;
216   
217 };
218 
219 
220 /**
221  * Remove the progress bar from the document.
222  */
223 X.progressbar.prototype.kill = function() {
224 
225   // hide the progress bar
226   if (this._style) {
227     goog.dom.removeNode(this._style);
228   }
229   if (this.getElement()) {
230     goog.dom.removeNode(this.getElement());
231   }
232   if (this._doneProgressBar) {
233     goog.dom.removeNode(this._doneProgressBar.getElement());
234   }
235   
236   this._style = null;
237   this._doneProgressBar = null;
238   
239 };
