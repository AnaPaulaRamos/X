 38 /**
 39  * Create a pair of shaders which consists of a vertex and a fragment shader.
 40  * 
 41  * @constructor
 42  * @extends X.base
 43  */
 44 X.shaders = function() {
 45 
 46   //
 47   // call the standard constructor of X.base
 48   goog.base(this);
 49   
 50   //
 51   // class attributes
 52   
 53   /**
 54    * @inheritDoc
 55    * @const
 56    */
 57   this._classname = 'shaders';
 58   
 59   /**
 60    * The vertex shader source of this shader pair. By default, a basic shader
 61    * supporting vertex positions and vertex colors is defined.
 62    * 
 63    * @type {!string}
 64    * @protected
 65    */
 66   this._vertexshaderSource = '';
 67   // The uncompiled vertex shader is in the file vertex.shader.
 68   // compilation via http://www.ctrl-alt-test.fr/minifier/index
 69   //
 70   // with the following settings:
 71   // GLSL, JS Header, .xyzw, Preserve external values
 72   //  
 73   var t = 'precision mediump float;'
 74       + "attribute vec3 vp,vn,vc;"
 75       + "attribute vec2 vtp;"
 76       + "attribute float vs;"
 77       + "uniform mat4 uv,up,uot;"
 78       + "uniform bool uuoc,uus,usrm;"
 79       + "uniform vec3 uc,uoc,usminc,usmaxc;"
 80       + "uniform float usmin,usmax,ups,usmint,usmaxt;"
 81       + "varying float fdn;"
 82       + "varying vec4 fvp;"
 83       + "varying vec2 ftp;"
 84       + "varying vec3 fc,fvn,ftvn;"
 85       + "void main()"
 86       + "{"
 87       + "ftvn=mat3(uv[0].xyz,uv[1].xyz,uv[2].xyz)*mat3(uot[0].xyz,uot[1].xyz,uot[2].xyz)*vn;"
 88       + "fvn=vn;" + "fdn=0.;" + "vec3 v=vp-uc;" + "fvp=uv*uot*vec4(v,1.);"
 89       + "ftp=vtp;" + "if(uus)" + "{" + "float f=vs;" + "if(f<usmint||f>usmaxt)"
 90       + "{" + "if(usrm)" + "fc=uoc;" + "else" + " fdn=1.;" + "}" + "else" + "{"
 91       + "if(usrm)" + "fc=f*usmaxc+(1.-f)*usminc;" + "else" + " fc=vc;" + "}"
 92       + "}" + "else" + " if(uuoc)" + "fc=uoc;" + "else" + " fc=vc;"
 93       + "gl_PointSize=ups;" + "gl_Position=up*fvp;" + "}";
 94   this._vertexshaderSource = t;
 95   
 96   /**
 97    * The fragment shader source of this shader pair. By default, a basic shader
 98    * supporting fragment colors is defined.
 99    * 
100    * @type {!string}
101    * @protected
102    */
103   this._fragmentshaderSource = '';
104   // The uncompiled fragment shader is in the file fragment.shader.
105   // compilation via http://www.ctrl-alt-test.fr/minifier/index
106   //
107   // with the following settings:
108   // GLSL, JS Header, .xyzw, Preserve external values
109   //  
110   var t2 = "precision mediump float;" + "uniform bool uup,uut,uutt,uulmt;"
111       + "uniform sampler2D uts,uts2;"
112       + "uniform float uoo,ulmo,uvlt,uvut,uvsmin,uvsmax;"
113       + "varying float fdn;" + "varying vec4 fvp;"
114       + "varying vec3 fc,fvn,ftvn;" + "varying vec2 ftp;" + "void main()" + "{"
115       + "if(fdn>0.)" + "{" + "discard;" + "}" + "if(uup)"
116       + "gl_FragColor=vec4(fc,1.);" + "else" + " if(uut)" + "{"
117       + "vec4 f=texture2D(uts,ftp),v=f;" + "if(uulmt)" + "{"
118       + "vec4 r=texture2D(uts2,ftp);" + "if(r.w>0.)" + "{" + "if(ulmo<1.)"
119       + "v=mix(r,v,1.-ulmo);" + "else" + " v=r;" + "}" + "}" + "if(uutt)" + "{"
120       + "float r=uvlt/uvsmax,l=uvut/uvsmax;" + "if(f.x<r||f.x>l)" + "{"
121       + "discard;" + "}" + "}" + "gl_FragColor=v;" + "gl_FragColor.w=uoo;"
122       + "}" + "else" + "{" + "vec3 f=normalize(ftvn);"
123       + "if(fvn==vec3(0.,0.,0.))" + "{" + "gl_FragColor=vec4(fc,1.);"
124       + "return;" + "}" + "vec3 v=vec3(0.,0.,1.),r=vec3(0,0,-10);"
125       + "r=normalize(r);" + "vec3 l=normalize(-fvp.xyz),u=reflect(-r,f);"
126       + "float d=pow(max(dot(u,l),0.),10.),o=.8*max(dot(f,v),0.),t=.3;"
127       + "gl_FragColor=vec4(fc*t+fc*o+vec3(.2,.2,.2)*d,uoo);" + "}" + "}";
128   this._fragmentshaderSource = t2;
129   
130 };
131 // inherit from X.base
132 goog.inherits(X.shaders, X.base);

193 X.shaders.prototype.vertex = function() {

212 /**
213  * Checks if this configured shaders object is valid in terms of using the
214  * defined attributes and uniforms.
215  * 
216  * @return {boolean} TRUE or FALSE depending on success.
217  * @throws {Error} An exception if the shader is invalid.
218  */
219 X.shaders.prototype.validate = function() {
220 
221   // check if the shader sources are compatible to the attributes and uniforms
222   
223   var attributes = Object.keys(X.shaders.attributes);
224   var uniforms = Object.keys(X.shaders.uniforms);
225   
226   // check if all attributes are used either in the vertex or the fragment
227   // shader
228   var attributesValid = attributes.every(function(a) {
229 
230     a = X.shaders.attributes[a];
231     return (this._vertexshaderSource.search(a) != -1) ||
232         (this._fragmentshaderSource.search(a) != -1);
233     
234   }.bind(this));
235   
236   if (!attributesValid) {
237     
238     throw new Error('Could not find all attributes in the shader sources.');
239     
240   }
241   
242   // check if all attributes are used either in the vertex or the fragment
243   // shader
244   var uniformsValid = uniforms.every(function(u) {
245 
246     u = X.shaders.uniforms[u];
247     return (this._vertexshaderSource.search(u) != -1) ||
248         (this._fragmentshaderSource.search(u) != -1);
249     
250   }.bind(this));
251   
252   if (!uniformsValid) {
253     
254     throw new Error('Could not find all uniforms in the shader sources.');
255     
256   }
257   
258   return true;
259   
260 };
