 38 /**
 39  * Create a 2D renderer inside a given DOM Element.
 40  * 
 41  * @constructor
 42  * @extends X.renderer
 43  */
 44 X.renderer2D = function() {
 45 
 46   //
 47   // call the standard constructor of X.renderer
 48   goog.base(this);
 49   
 50   //
 51   // class attributes
 52   
 53   /**
 54    * @inheritDoc
 55    * @const
 56    */
 57   this._classname = 'renderer2D';
 58   
 59   /**
 60    * The orientation of this renderer.
 61    * 
 62    * @type {?string}
 63    * @protected
 64    */
 65   this._orientation = null;
 66   
 67   /**
 68    * A frame buffer for slice data.
 69    * 
 70    * @type {?Element}
 71    * @protected
 72    */
 73   this._frameBuffer = null;
 74   
 75   /**
 76    * The rendering context of the slice frame buffer.
 77    * 
 78    * @type {?Object}
 79    * @protected
 80    */
 81   this._frameBufferContext = null;
 82   
 83   /**
 84    * A frame buffer for label data.
 85    * 
 86    * @type {?Element}
 87    * @protected
 88    */
 89   this._labelFrameBuffer = null;
 90   
 91   /**
 92    * The rendering context of the label frame buffer.
 93    * 
 94    * @type {?Object}
 95    * @protected
 96    */
 97   this._labelFrameBufferContext = null;
 98   
 99   /**
100    * The current slice width.
101    * 
102    * @type {number}
103    * @protected
104    */
105   this._sliceWidth = 0;
106   
107   /**
108    * The current slice height.
109    * 
110    * @type {number}
111    * @protected
112    */
113   this._sliceHeight = 0;
114   
115 };
116 // inherit from X.base
117 goog.inherits(X.renderer2D, X.renderer);

126 X.renderer2D.prototype.onScroll = function() {
127 
128   // do nothing

132 /**
133  * @inheritDoc
134  */
135 X.renderer2D.prototype.onScroll_ = function(event) {
136 
137   goog.base(this, 'onScroll_', event);
138   
139   // grab the current volume
140   var _volume = this._topLevelObjects[0];
141   // .. if there is none, exit right away
142   if (!_volume) {
143     return;
144   }
145   
146   // switch between different orientations
147   var _orientation = this._orientation;
148   
149   if (event._up) {
150     
151     // yes, scroll up
152     _volume['index' + _orientation] = _volume['index' + _orientation] + 1;
153     
154   } else {
155     
156     // yes, so scroll down
157     _volume['index' + _orientation] = _volume['index' + _orientation] - 1;
158     
159   }
160   
161   _volume.modified();
162   
163   // execute the callback
164   eval('this.onScroll();');
165   
166   // .. and trigger re-rendering
167   this.render_(false, false);
168   
169 };

172 /**
173  * Get the orientation of this renderer. Valid orientations are 'x','y','z' or
174  * null.
175  * 
176  * @return {?string} The orientation of this renderer.
177  */
178 X.renderer2D.prototype.__defineGetter__('orientation', function() {
179 
180   return this._orientation;
181   
182 });
183 
184 
185 /**
186  * Set the orientation for this renderer. Valid orientations are 'x','y' or 'z'.
187  * 
188  * @param {!string} orientation The orientation for this renderer: 'x','y' or
189  *          'z'.
190  * @throws {Error} An error, if the given orientation was wrong.
191  */
192 X.renderer2D.prototype.__defineSetter__('orientation', function(orientation) {
193 
194   orientation = orientation.toUpperCase();
195   
196   if (orientation != 'X' && orientation != 'Y' && orientation != 'Z') {
197     
198     throw new Error('Invalid orientation.');
199     
200   }
201   
202   this._orientation = orientation;
203   
204 });

207 /**
208  * @inheritDoc
209  */
210 X.renderer2D.prototype.init = function() {
211 
212   // make sure an orientation is configured
213   if (!this._orientation) {
214     
215     throw new Error('No 2D orientation set.');
216     
217   }
218   
219   // call the superclass' init method
220   goog.base(this, 'init', '2d');
221   
222   // use the background color of the container by setting transparency here
223   this._context.fillStyle = "rgba(0,0,0,0)";
224   // .. and size
225   this._context.fillRect(0, 0, this._canvas.width, this._canvas.height);
226   
227   // create an invisible canvas as a framebuffer
228   this._frameBuffer = goog.dom.createDom('canvas');
229   this._labelFrameBuffer = goog.dom.createDom('canvas');
230   //
231   // 
232   // try to apply nearest-neighbor interpolation -> does not work right now
233   // so we ignore it
234   // this._labelFrameBuffer.style.imageRendering = 'optimizeSpeed';
235   // this._labelFrameBuffer.style.imageRendering = '-moz-crisp-edges';
236   // this._labelFrameBuffer.style.imageRendering = '-o-crisp-edges';
237   // this._labelFrameBuffer.style.imageRendering = '-webkit-optimize-contrast';
238   // this._labelFrameBuffer.style.imageRendering = 'optimize-contrast';
239   // this._labelFrameBuffer.style.msInterpolationMode = 'nearest-neighbor';
240   
241 };

248 
249   // call the super class

251   
252   // .. and perform auto scaling
253   this.autoScale_();
254   // .. render
255   this.render_(false, false);

260 /**
261  * @inheritDoc
262  */
263 X.renderer2D.prototype.update_ = function(object) {
264 
265   // call the update_ method of the superclass
266   goog.base(this, 'update_', object);
267   
268   // check if object already existed..
269   var existed = false;
270   
271   if (this.get(object._id)) {
272     // this means, we are updating
273     existed = true;
274     
275   }
276   
277   // var id = object._id;
278   // var texture = object._texture;
279   var file = object._file;
280   var labelmap = object._labelmap; // here we access directly since we do not
281   // want to create one using the labelmap() singleton accessor
282   var colortable = object._colortable;
283   
284   //
285   // LABEL MAP
286   //
287   if (goog.isDefAndNotNull(labelmap) && goog.isDefAndNotNull(labelmap._file) &&
288       labelmap._file._dirty) {
289     // a labelmap file is associated to this object and it is dirty..
290     // background: we always want to parse label maps first
291     
292     // run the update_ function on the labelmap object
293     this.update_(labelmap);
294     
295     // jump out
296     return;
297     
298   }
299   
300   //
301   // COLOR TABLE
302   //
303   if (goog.isDefAndNotNull(colortable) &&
304       goog.isDefAndNotNull(colortable._file) && colortable._file._dirty) {
305     // a colortable file is associated to this object and it is dirty..
306     
307     // start loading
308     this._loader.load(colortable, object);
309     
310     return;
311     
312   }
313   
314   //
315   // VOLUME
316   //
317   if (goog.isDefAndNotNull(file) && file._dirty) {
318     // this object is based on an external file and it is dirty..
319     
320     // start loading..
321     this._loader.load(object, object);
322     
323     return;
324     
325   }
326   
327   //
328   // at this point the orientation of this renderer might have changed so we
329   // should recalculate all the cached values
330   
331   var _sliceWidth = 0;
332   var _sliceHeight = 0;
333   var _dimensions = object._dimensions;
334   
335   // check the orientation and store a pointer to the slices
336   if (this._orientation == 'X') {
337     
338     this._slices = object._slicesX._children;
339     _sliceWidth = _dimensions[2];
340     _sliceHeight = _dimensions[1];
341     
342   } else if (this._orientation == 'Y') {
343     
344     this._slices = object._slicesY._children;
345     _sliceWidth = _dimensions[0];
346     _sliceHeight = _dimensions[2];
347     
348   } else if (this._orientation == 'Z') {
349     
350     this._slices = object._slicesZ._children;
351     _sliceWidth = _dimensions[0];
352     _sliceHeight = _dimensions[1];
353     
354   }
355   
356   if (this._orientation == 'X') {
357     
358     // the X oriented texture is twisted ..
359     var _newSliceWidth = _sliceHeight;
360     _sliceHeight = _sliceWidth;
361     _sliceWidth = _newSliceWidth;
362     
363   }
364   // .. and store the dimensions
365   this._sliceWidth = _sliceWidth;
366   this._sliceHeight = _sliceHeight;
367   
368   // update the invisible canvas to store the current slice
369   var _frameBuffer = this._frameBuffer;
370   _frameBuffer.width = _sliceWidth;
371   _frameBuffer.height = _sliceHeight;
372   var _frameBuffer2 = this._labelFrameBuffer;
373   _frameBuffer2.width = _sliceWidth;
374   _frameBuffer2.height = _sliceHeight;
375   // .. and the context
376   this._frameBufferContext = _frameBuffer.getContext('2d');
377   this._labelFrameBufferContext = _frameBuffer2.getContext('2d');
378   
379 
380 
381   // do the following only if the object is brand-new
382   if (!existed) {
383     this._objects.add(object);
384     this.autoScale_();
385   }
386   
387 };

390 /**
391  * Adjust the zoom (scale) to best fit the current slice.
392  */
393 X.renderer2D.prototype.autoScale_ = function() {
394 
395   // let's auto scale for best fit
396   var _wScale = this._width / this._sliceWidth;
397   var _hScale = this._height / this._sliceHeight;
398   
399   var _autoScale = Math.min(_wScale, _hScale);
400   _autoScale = 10 * _autoScale - 10;
401   
402   // propagate scale (zoom) to the camera
403   var _view = this._camera._view;
404   _view.setValueAt(2, 3, _autoScale);
405   
406 };

409 /**
410  * @inheritDoc
411  */
412 X.renderer2D.prototype.render_ = function(picking, invoked) {
413 
414   // call the render_ method of the superclass
415   goog.base(this, 'render_', picking, invoked);
416   
417   // only proceed if there are actually objects to render
418   var _objects = this._objects.values();
419   var _numberOfObjects = _objects.length;
420   if (_numberOfObjects == 0) {
421     // there is nothing to render
422     // get outta here
423     return;
424   }
425   
426   //
427   // grab the camera settings
428   //
429   
430   // viewport size
431   var _width = this._width;
432   var _height = this._height;
433   
434   // first grab the view matrix which is 4x4 in favor of the 3D renderer
435   var _view = this._camera._view;
436   
437   // clear the canvas
438   this._context.save();
439   this._context.clearRect(-_width, -_height, 2 * _width, 2 * _height);
440   this._context.restore();
441   
442   // transform the canvas according to the view matrix
443   var _x = 1 * _view.getValueAt(0, 3);
444   var _y = -1 * _view.getValueAt(1, 3); // we need to flip y here
445   // .. this includes zoom
446   var _normalizedScale = Math.max(1 + _view.getValueAt(2, 3) / 10, 0.6);
447   this._context.setTransform(_normalizedScale, 0, 0, _normalizedScale, _x, _y);
448   
449 
450   //
451   // grab the volume and current slice
452   //
453   var _volume = this._topLevelObjects[0];
454   var _currentSlice = _volume['index' + this._orientation];
455   
456   // .. here is the current slice
457   var _slice = this._slices[parseInt(_currentSlice, 10)];
458   var _sliceData = _slice._texture._rawData;
459   var _currentLabelMap = _slice._labelmap;
460   var _labelData = null;
461   if (_currentLabelMap) {
462     _labelData = _currentLabelMap._rawData;
463   }
464   var _sliceWidth = this._sliceWidth;
465   var _sliceHeight = this._sliceHeight;
466   
467   // canvas center, taking zooming and panning in account
468   var _canvasCenterX = (1 / (_normalizedScale * 2)) *
469       (_width - _sliceWidth * _normalizedScale) + _x;
470   var _canvasCenterY = (1 / (_normalizedScale * 2)) *
471       (_height - _sliceHeight * _normalizedScale) + _y;
472   
473 
474   //
475   // FRAME BUFFERING
476   //  
477   var _imageFBContext = this._frameBufferContext;
478   var _labelFBContext = this._labelFrameBufferContext;
479   
480   // grab the current pixels
481   var _imageData = _imageFBContext
482       .getImageData(0, 0, _sliceWidth, _sliceHeight);
483   var _labelmapData = _labelFBContext.getImageData(0, 0, _sliceWidth,
484       _sliceHeight);
485   var _pixels = _imageData.data;
486   var _labelPixels = _labelmapData.data;
487   var _pixelsLength = _pixels.length;
488   
489   // threshold values
490   var _maxScalarRange = _volume._max;
491   var _lowerThreshold = _volume._lowerThreshold;
492   var _upperThreshold = _volume._upperThreshold;
493   
494 
495   // loop through the pixels and draw them to the invisible canvas
496   // from bottom right up
497   // also apply thresholding
498   var _index = 0;
499   do {
500     
501     // default color and label is just transparent
502     var _color = [0, 0, 0, 0];
503     var _label = [0, 0, 0, 0];
504     
505     // grab the pixel intensity
506     var _intensity = _sliceData[_index] / 255 * _maxScalarRange;
507     
508     // apply thresholding
509     if (_intensity >= _lowerThreshold && _intensity <= _upperThreshold) {
510       
511       // current intensity is inside the threshold range so use the real
512       // intensity
513       _color = [_sliceData[_index], _sliceData[_index + 1],
514                 _sliceData[_index + 2], _sliceData[_index + 3]];
515       
516       if (_currentLabelMap) {
517         
518         // we have a label map here
519         _label = [_labelData[_index], _labelData[_index + 1],
520                   _labelData[_index + 2], _labelData[_index + 3]];
521         
522       }
523       
524     }
525     
526     var _invertedIndex = (_pixelsLength - 1 - _index);
527     
528     _pixels[_invertedIndex - 3] = _color[0]; // r
529     _pixels[_invertedIndex - 2] = _color[1]; // g
530     _pixels[_invertedIndex - 1] = _color[2]; // b
531     _pixels[_invertedIndex] = _color[3]; // a
532     
533 
534     _labelPixels[_invertedIndex - 3] = _label[0]; // r
535     _labelPixels[_invertedIndex - 2] = _label[1]; // g
536     _labelPixels[_invertedIndex - 1] = _label[2]; // b
537     _labelPixels[_invertedIndex] = _label[3]; // a
538     
539 
540     _index = _index + 4; // increase by 4 units for r,g,b,a
541     
542   } while (_index < _pixelsLength);
543   
544   // store the generated image data to the frame buffer context
545   _imageFBContext.putImageData(_imageData, 0, 0);
546   _labelFBContext.putImageData(_labelmapData, 0, 0);
547   
548 
549 
550   //
551   // the actual drawing (rendering) happens here
552   //
553   
554   // draw the slice frame buffer (which equals the slice data) to the main
555   // context
556   this._context.globalAlpha = 1.0; // draw fully opaque
557   this._context.drawImage(this._frameBuffer, _canvasCenterX, _canvasCenterY);
558   
559   // draw the labels with a configured opacity
560   if (_currentLabelMap && _volume._labelmap._visible) {
561     
562     var _labelOpacity = _volume._labelmap._opacity;
563     
564     this._context.globalAlpha = _labelOpacity; // draw transparent depending on
565     // opacity
566     this._context.drawImage(this._labelFrameBuffer, _canvasCenterX,
567         _canvasCenterY);
568   }
569   
570 };
