 30 goog.provide('X.texture');
 31 
 32 // requires
 33 goog.require('X.base');
 34 goog.require('X.loadable');
 35 
 36 
 37 
 38 /**
 39  * Create a texture using a two-dimensional image file or using raw-data.
 40  * 
 41  * @constructor
 42  * @extends X.base
 43  */
 44 X.texture = function() {
 45 
 46   //
 47   // call the standard constructor of X.base
 48   goog.base(this);
 49   
 50   //
 51   // class attributes
 52   
 53   /**
 54    * @inheritDoc
 55    * @const
 56    */
 57   this._classname = 'texture';
 58   
 59   /**
 60    * The file of this texture.
 61    * 
 62    * @type {?X.file}
 63    * @protected
 64    */
 65   this._file = null;
 66   
 67   /**
 68    * The (optional) HTML Image element.
 69    * 
 70    * @type {?Element}
 71    * @protected
 72    */
 73   this._image = null;
 74   
 75   /**
 76    * The raw data of this texture as a uint8 array.
 77    * 
 78    * @type {?Object}
 79    * @protected
 80    */
 81   this._rawData = null;
 82   
 83   /**
 84    * The width of the raw data.
 85    * 
 86    * @type {number}
 87    * @protected
 88    */
 89   this._rawDataWidth = 0;
 90   
 91   /**
 92    * The height of the raw data.
 93    * 
 94    * @type {number}
 95    * @protected
 96    */
 97   this._rawDataHeight = 0;
 98   
 99   // inject functionality
100   inject(this, new X.loadable()); // this object is loadable from a file
101   
102 };
103 // inherit from X.base
104 goog.inherits(X.texture, X.base);
105 
106 goog.exportSymbol('X.texture', X.texture);
