 43 /**
 44  * Create a displayable object. Objects may have points, colors, a texture in
 45  * addition to opacity and visibility settings. If another X.object is passed to
 46  * this constructor, the properties from this X.object are used to configure the
 47  * new one.
 48  * 
 49  * @constructor
 50  * @param {X.object=} object Another X.object to use as a template.
 51  * @extends X.base
 52  */
 53 X.object = function(object) {
 54 
 55   //
 56   // call the standard constructor of X.base
 57   goog.base(this);
 58   
 59   //
 60   // class attributes
 61   
 62   /**
 63    * @inheritDoc
 64    * @const
 65    */
 66   this._classname = 'object';
 67   
 68   /**
 69    * The children of this object.
 70    * 
 71    * @type {!Array}
 72    * @protected
 73    */
 74   this._children = new Array();
 75   
 76   /**
 77    * The color table of this object.
 78    * 
 79    * @type {?X.colortable}
 80    * @protected
 81    */
 82   this._colortable = null;
 83   
 84   /**
 85    * The scalars of this object.
 86    * 
 87    * @type {?X.scalars}
 88    * @protected
 89    */
 90   this._scalars = null;
 91   
 92   // inject functionality
 93   inject(this, new X.displayable()); // this object is displayable
 94   
 95   if (goog.isDefAndNotNull(object)) {
 96     
 97     // copy the properties of the given object over
 98     this.copy_(object);
 99     
100   }
101   
102 };
103 // inherit from X.base
104 goog.inherits(X.object, X.base);

107 /**
108  * Copies the properties from a given object to this object. The texture,
109  * textureCoordinateMap and the children are not copied but linked.
110  * 
111  * @param {!X.object} object The given object.
112  * @protected
113  */
114 X.object.prototype.copy_ = function(object) {
115 
116   this._type = object._type;
117   
118   this._transform = new X.transform();
119   this._transform._matrix = new X.matrix(object._transform._matrix.array_);
120   
121   this._color = object._color.slice();
122   
123   this._points = new X.triplets(object._points);
124   
125   this._normals = new X.triplets(object._normals);
126   
127   this._colors = new X.triplets(object._colors);
128   
129   // do we need to copy this? maybe not
130   this._texture = object._texture;
131   this._textureCoordinateMap = object._textureCoordinateMap;
132   
133   if (object._file) {
134     // only if a file is configured
135     this._file = new X.file(new String(object._file._path).toString());
136   }
137   
138   this._opacity = object._opacity;
139   
140   // 
141   // children
142   this._children.length = 0; // remove old ones
143   var _oldChildren = object._children;
144   if (_oldChildren) {
145     var _oldChildrenLength = _oldChildren.length;
146     var i = 0;
147     for (i = 0; i < _oldChildrenLength; i++) {
148       
149       // dynamic duck typing
150       var classname = _oldChildren[i]._classname;
151       this._children.push(new X[classname](_oldChildren[i]));
152       
153     }
154   }
155   
156   this._visible = object._visible;
157   
158   this._pointsize = object._pointsize;
159   
160   this._linewidth = object._linewidth;
161   
162   if (object._caption) {
163     // only if a caption is configured
164     this._caption = new String(object._caption).toString();
165   }
166   
167   this._magicmode = object._magicmode;
168   
169   this._pointIndices = object._pointIndices.slice();
170   
171   this._dirty = true;
172   
173 };

237 /**
238  * Compare two X.objects by their opacity values and their distance to the
239  * viewer's eye. Fully opaque objects should be always ordered before
240  * transparent ones, and the transparent ones should be ordered back-to-front in
241  * terms of the distance to the viewer's eye.
242  * 
243  * @param {X.object} object1 Object1 to compare against Object2.
244  * @param {X.object} object2 Object2 to compare against Object1.
245  * @return {!number} 1, if Object1 should be ordered after Object2. -1, if
246  *         Object1 should be ordered before Object2
247  */
248 X.object.OPACITY_COMPARATOR = function(object1, object2) {
249 
250   // check if we have two valid objects to compare
251   if (!goog.isDefAndNotNull(object1) || !goog.isDefAndNotNull(object2) ||
252       !(object1 instanceof X.object) || !(object2 instanceof X.object)) {
253     
254     throw new Error('Fatal: Two valid X.objects are required for comparison.');
255     
256   }
257   
258   // full opaque objects should always be rendered first
259   if (object1._opacity == 1) {
260     
261     // always put object1 before object2
262     return -1;
263     
264   }
265   if (object2._opacity == 1) {
266     
267     // always put object2 before object1
268     return 1;
269     
270   }
271   
272   if (goog.isDefAndNotNull(object1._distance) &&
273       goog.isDefAndNotNull(object2._distance)) {
274     
275     // order back-to-front from the viewer's eye
276     
277     if (object1._distance > object2._distance) {
278       
279       // object2 is closer so object1 should be ordered (drawn) before object2
280       return -1;
281       
282     } else if (object1._distance <= object2._distance) {
283       
284       // object 1 is closer so object1 should be ordered (drawn) after object2
285       return 1;
286       
287     }
288     
289 
290   }
291   
292   return 1;
293   
294 };
