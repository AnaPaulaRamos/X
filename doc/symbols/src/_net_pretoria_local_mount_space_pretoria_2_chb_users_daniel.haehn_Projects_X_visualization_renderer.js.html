 58 /**
 59  * The superclass for all renderers.
 60  * 
 61  * @constructor
 62  * @extends X.base
 63  */
 64 X.renderer = function() {
 65 
 66   //
 67   // call the standard constructor of X.base
 68   goog.base(this);
 69   
 70   /**
 71    * @inheritDoc
 72    * @const
 73    */
 74   this.__classname = 'renderer';
 75   
 76   /**
 77    * The HTML container of this renderer, E.g. a <div>.
 78    * 
 79    * @type {!Element|HTMLBodyElement}
 80    * @public
 81    */
 82   this._container = window.document.body;
 83   
 84   /**
 85    * The width of this renderer.
 86    * 
 87    * @type {!number}
 88    * @public
 89    */
 90   this._width = this._container.clientWidth;
 91   
 92   /**
 93    * The height of this renderer.
 94    * 
 95    * @type {!number}
 96    * @public
 97    */
 98   this._height = this._container.clientHeight;
 99   
100   /**
101    * The Canvas of this renderer.
102    * 
103    * @type {?Element}
104    * @public
105    */
106   this._canvas = null;
107   
108   /**
109    * The camera of this renderer.
110    * 
111    * @type {?X.camera}
112    * @protected
113    */
114   this._camera = null;
115   
116   /**
117    * The interactor of this renderer.
118    * 
119    * @type {?X.interactor}
120    * @protected
121    */
122   this._interactor = null;
123   
124   /**
125    * An X.array containing the displayable objects of this renderer. The object
126    * reflects the rendering order for the associated objects.
127    * 
128    * @type {!X.array}
129    * @protected
130    */
131   this._objects = new X.array(X.object.OPACITY_COMPARATOR);
132   
133   /**
134    * An array containing the topLevel objects (which do not have parents) of
135    * this renderer.
136    * 
137    * @type {!Array}
138    * @protected
139    */
140   this._topLevelObjects = new Array();
141   
142   /**
143    * The loader associated with this renderer.
144    * 
145    * @type {?X.loader}
146    * @protected
147    */
148   this._loader = null;
149   
150   /**
151    * A locked flag for synchronizing.
152    * 
153    * @type {boolean}
154    * @protected
155    */
156   this._locked = false;
157   
158   /**
159    * A flag to show if the initial loading was completed.
160    * 
161    * @type {boolean}
162    * @public
163    */
164   this._loadingCompleted = false;
165   
166   /**
167    * The progressBar of this renderer.
168    * 
169    * @type {?X.progressbar}
170    * @protected
171    */
172   this._progressBar = null;
173   
174   /**
175    * The rendering context of this renderer.
176    * 
177    * @type {?Object}
178    * @protected
179    */
180   this._context = null;
181   
182   /**
183    * The configuration of this renderer.
184    * 
185    * @enum {boolean}
186    */
187   this['config'] = {
188     'PROGRESSBAR_ENABLED': true
189   };
190   
191   window.console.log('XTK Release 4 -- 04/12/12 -- http://www.goXTK.com');
192   
193 };

330 /**
331  * Set the container for this renderer. This has to happen before
332  * X.renderer.init() is called.
333  * 
334  * @param {!string|Element|HTMLBodyElement} container Either an ID to a DOM
335  *          container or the DOM element itself.
336  * @throws {Error} An error, if the given container is invalid.
337  * @public
338  */
339 X.renderer.prototype.__defineSetter__('container', function(container) {
340 
341   // check if a container is passed
342   if (!goog.isDefAndNotNull(container)) {
343     
344     throw new Error('An ID to a valid container (<div>..) is required.');
345     
346   }
347   
348   // check if the passed container is really valid
349   var _container = container;
350   
351   // if an id is given, try to get the corresponding DOM element
352   if (goog.isString(_container)) {
353     
354     _container = goog.dom.getElement(container);
355     
356   }
357   
358   // now we should have a valid DOM element
359   if (!goog.dom.isElement(_container)) {
360     
361     throw new Error('Could not find the given container.');
362     
363   }
364   
365   this._container = _container;
366   
367 });

390   // only do the following if the progressBar was not turned off

394     // loader is working

406 /**
407  * Hides the loading progress bar.
408  * 
409  * @protected
410  */
411 X.renderer.prototype.hideProgressBar_ = function() {
412 
413   // only do the following if the progressBar was not turned off
414   if (this['config']['PROGRESSBAR_ENABLED']) {
415     
416     if (this._progressBar && !this.__readyCheckTimer2) {
417       
418       // show a green, full progress bar
419       this._progressBar.done();
420       
421       // wait for a short time
422       this.__readyCheckTimer2 = goog.Timer.callOnce(function() {
423 
424         this.__readyCheckTimer2 = null;
425         
426         if (this._progressBar) {
427           
428           // we are done, kill the progressbar
429           this._progressBar.kill();
430           this._progressBar = null;
431           
432         }
433         
434         this.render();
435         
436       }.bind(this), 700);
437       // .. and jump out
438       return;
439       
440     } // if progressBar still exists
441     
442   } // if progressBar is enabled
443   
444 };

447 /**
448  * Create the canvas of this renderer inside the configured container and using
449  * attributes like width, height etc. Then, initialize the rendering context and
450  * attach all necessary objects (e.g. camera, shaders..). Finally, initialize
451  * the event listeners.
452  * 
453  * @param {string} _contextName The name of the context to create.
454  * @throws {Error} An exception if there were problems during initialization.
455  * @protected
456  */
457 X.renderer.prototype.init = function(_contextName) {
458 
459   // create the canvas
460   var _canvas = goog.dom.createDom('canvas');
461   
462   //
463   // append it to the container
464   goog.dom.appendChild(this._container, _canvas);
465   
466   // the container might have resized now, so update our width and height
467   // settings
468   this._width = this._container.clientWidth;
469   this._height = this._container.clientHeight;
470   
471   // width and height can not be set using CSS but via object properties
472   _canvas.width = this._width;
473   _canvas.height = this._height;
474   
475 
476   // --------------------------------------------------------------------------
477   //
478   // Viewport initialization
479   //
480   
481   //
482   // Step1: Get Context of canvas
483   //
484   try {
485     
486     var _context = _canvas.getContext(_contextName);
487     
488     if (!_context) {
489       
490       // this exception triggers the display of the error message
491       // because the context creation can either fail with an exception
492       // or return a NULL context
493       throw new Error();
494       
495     }
496     
497   } catch (e) {
498     
499     // Canvas2D is not supported with this browser/machine/gpu
500     
501     // attach a message to the container's inner HTML
502     var _style = "color:red;font-family:sans-serif;";
503     var _msg = 'Sorry, ' +
504         _contextName +
505         ' context is <strong>not supported</strong> on this machine! See <a href="http://crash.goXTK.com" target="_blank">http://crash.goXTK.com</a> for requirements..';
506     this._container.innerHTML = '<h3 style="' + _style +
507         '">Oooops..</h3><p style="' + _style + '">' + _msg + '</p>';
508     
509     // .. and throw an exception
510     throw new Error(_msg);
511     
512   }
513   
514   //
515   // Step 1b: Configure the X.loader
516   //
517   this._loader = new X.loader();
518   
519   // listen to a progress event which gets fired during loading whenever
520   // progress was made
521   goog.events.listen(this._loader, X.event.events.PROGRESS, this.onProgress
522       .bind(this));
523   
524   //
525   // Step 1c: Register the created canvas to this instance
526   //
527   this._canvas = _canvas;
528   
529   //
530   // Step 1d: Register the created context to this instance
531   //
532   this._context = _context;
533   
534   //
535   // Step2: Configure the context and the viewport
536   //
537   
538   //
539   // create a new interactor
540   var _interactor = new X.interactor3D(this._canvas);
541   
542   // in the 2d case, create a 2d interactor (of course..)
543   if (_contextName == '2d') {
544     
545     _interactor = new X.interactor2D(this._canvas);
546     
547   }
548   // initialize it and..
549   _interactor.init();
550   
551   // .. listen to resetViewEvents
552   goog.events.listen(_interactor, X.event.events.RESETVIEW,
553       this.resetViewAndRender.bind(this));
554   // .. listen to hoverEvents
555   goog.events.listen(_interactor, X.event.events.HOVER, this.onHover_
556       .bind(this));
557   // .. listen to scroll events
558   goog.events.listen(_interactor, X.event.events.SCROLL, this.onScroll_
559       .bind(this));
560   
561 
562   // .. and finally register it to this instance
563   this._interactor = _interactor;
564   
565   //
566   // create a new camera
567   // width and height are required to calculate the perspective
568   var _camera = new X.camera3D(this._width, this._height);
569   
570   if (_contextName == '2d') {
571     _camera = new X.camera2D(this._width, this._height);
572   }
573   // observe the interactor for user interactions (mouse-movements etc.)
574   _camera.observe(this._interactor);
575   // ..listen to render requests from the camera
576   // these get fired after user-interaction and camera re-positioning to re-draw
577   // all objects
578   goog.events.listen(_camera, X.event.events.RENDER, this.render_.bind(this,
579       false, false));
580   
581   //
582   // attach all created objects as class attributes
583   // should be one of the last things to do here since we use these attributes
584   // to check if the initialization was completed successfully
585   this._camera = _camera;
586   
587 
588   //
589   //
590   // .. the rest should be performed in the subclasses
591   
592 };

595 /**
596  * Add a new object to this renderer. The renderer has to be initialized before
597  * doing so. A X.renderer.render() call has to be initiated to display added
598  * objects.
599  * 
600  * @param {!X.object} object The object to add to this renderer.
601  * @throws {Error} An exception if something goes wrong.
602  */
603 X.renderer.prototype.add = function(object) {
604 
605   // for constructable objects (e.g. cube, sphere, cylinder), we call the
606   // modified() function to generate the CSG representations
607   if (object instanceof X.cube || object instanceof X.sphere ||
608       object instanceof X.cylinder) {
609     
610     object.modified();
611     
612   }
613   
614   // we know that objects which are directly added using this function are def.
615   // top-level objects, meaning that they do not have a parent
616   this._topLevelObjects.push(object);
617   
618   this.update_(object);
619   
620 };

623 /**
624  * Configure a displayable object within this renderer. The object can be a
625  * newly created one or an existing one. A X.renderer.render() call has to be
626  * initiated to display the object.
627  * 
628  * @param {!X.object} object The displayable object to setup within this
629  *          renderer.
630  * @throws {Error} An exception if something goes wrong.
631  * @protected
632  */
633 X.renderer.prototype.update_ = function(object) {
634 
635   if (!this._canvas || !this._context) {
636     
637     throw new Error('The renderer was not initialized properly.');
638     
639   }
640   
641   if (!goog.isDefAndNotNull(object)) {
642     window.console.log(object);
643     throw new Error('Illegal object.');
644     
645   }
646   
647   // listen to modified events of this object, if we didn't do that before
648   if (!goog.events.hasListener(object, X.event.events.MODIFIED)) {
649     
650     goog.events.listen(object, X.event.events.MODIFIED, this.onModified
651         .bind(this));
652     
653   }
654   
655 };

658 /**
659  * Get the existing X.object with the given id.
660  * 
661  * @param {!number} id The object's id.
662  * @return {?X.object} The requested X.object or null if it was not found.
663  * @throws {Error} If the given id was invalid.
664  * @public
665  */
666 X.renderer.prototype.get = function(id) {
667 
668   if (!goog.isDefAndNotNull(id)) {
669     
670     throw new Error('Invalid object id.');
671     
672   }
673   
674   // loop through objects and try to find the id
675   var _objects = this._objects.values();
676   var _numberOfObjects = _objects.length;
677   
678   var _k = 0;
679   for (_k = 0; _k < _numberOfObjects; _k++) {
680     
681     if (_objects[_k]._id == id) {
682       
683       // found!
684       return _objects[_k];
685       
686     }
687     
688   }
689   
690   // not found
691   return null;
692   
693 };

717 /**
718  * Recursively loop through a hierarchy tree of objects and print it.
719  * 
720  * @param {!X.object} object The starting point object.
721  * @param {number} level The current level in the scene hierarchy.
722  * @protected
723  */
724 X.renderer.prototype.generateTree_ = function(object, level) {
725 
726   var _output = "";
727   
728   var _l = 0;
729   for (_l = 0; _l < level; _l++) {
730     
731     _output += ">";
732     
733   }
734   
735   _output += object._id;
736   
737   window.console.log(_output);
738   
739   if (object._children.length > 0) {
740     
741     // loop through the children
742     var _children = object._children;
743     var _numberOfChildren = _children.length;
744     var _c = 0;
745     
746     for (_c = 0; _c < _numberOfChildren; _c++) {
747       
748       this.generateTree_(_children[_c], level + 1);
749       
750     }
751     
752   }
753

757 /**
758  * (Re-)render all associated displayable objects of this renderer. This method
759  * clears the viewport and re-draws everything by looping through the tree of
760  * objects. The current camera is used to setup the world space.
761  * 
762  * @public
763  */
764 X.renderer.prototype.render = function() {
765 
766   if (!this._canvas || !this._context) {
767     
768     throw new Error('The renderer was not initialized properly.');
769     
770   }
771   
772   // READY CHECK
773   //
774   // now we check if we are ready to display everything
775   // - ready means: all textures loaded and setup, all external files loaded and
776   // setup and all other objects loaded and setup
777   //
778   // if we are not ready, we wait..
779   // if we are ready, we continue with the rendering
780   
781   // let's check if render() was called before and the single-shot timer is
782   // already there
783   // f.e., if we are in a setInterval-configured render loop, we do not want to
784   // create multiple single-shot timers
785   if (goog.isDefAndNotNull(this._readyCheckTimer)) {
786     
787     return;
788     
789   }
790   
791   //
792   // LOADING..
793   //
794   if (!this._loader.completed()) {
795     
796     // we are not ready yet.. the loader is still working;
797     
798     this.showProgressBar_();
799     
800     // let's check again in a short time
801     this._readyCheckTimer = goog.Timer.callOnce(function() {
802 
803       this._readyCheckTimer = null; // destroy the timer
804       
805       // try to render now..
806       // if the loader is ready it will work, else wise another single-shot gets
807       // configured in 500 ms
808       this.render();
809       
810     }.bind(this), 100); // check again in 500 ms
811     
812     return; // .. and jump out
813     
814   } else {
815     
816     // we are ready! yahoooo!
817     
818     // call the onShowtime function which can be overloaded
819     if (!this._loadingCompleted) {
820       
821       eval("this.onShowtime()");
822       this._loadingCompleted = true; // flag the renderer as 'initial
823       // loading completed'
824       
825     }
826     
827     // if we have a progress bar
828     if (this._progressBar) {
829       
830       // this means the X.loader is done..
831       this.hideProgressBar_();
832       
833       // .. we exit here since the hiding takes some time and automatically
834       // triggers the rendering when done
835       return;
836       
837     }
838     
839   }
840   //
841   // END OF LOADING
842   //
843   
844   //
845   // CURTAIN UP! LET THE SHOW BEGIN..
846   //
847   this.render_(false, true);
848   
849 };
