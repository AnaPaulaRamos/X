 31 goog.provide('X.scalars');

 40 /**
 41  * Create a container for a scalar array.
 42  * 
 43  * @constructor
 44  * @extends X.base
 45  */
 46 X.scalars = function() {
 47 
 48   //
 49   // call the standard constructor of X.base
 50   goog.base(this);
 51   
 52   //
 53   // class attributes
 54   
 55   /**
 56    * @inheritDoc
 57    * @const
 58    */
 59   this._classname = 'scalars';
 60   
 61   /**
 62    * The array containing the scalars.
 63    * 
 64    * @type {?Array}
 65    * @protected
 66    */
 67   this._array = null;
 68   
 69   /**
 70    * The WebGL-ready version of the scalars array.
 71    * 
 72    * @type {?Array}
 73    * @protected
 74    */
 75   this._glArray = null;
 76   
 77   /**
 78    * The color to map the min. scalar.
 79    * 
 80    * @type {!Array}
 81    * @protected
 82    */
 83   this._minColor = [0, 1, 0];
 84   
 85   /**
 86    * The color to map the max. scalar.
 87    * 
 88    * @type {!Array}
 89    * @protected
 90    */
 91   this._maxColor = [1, 0, 0];
 92   
 93   /**
 94    * Flag to replace the colors after thresholding. If FALSE, discard the
 95    * vertex.
 96    * 
 97    * @type {boolean}
 98    * @protected
 99    */
100   this._replaceMode = true;
101   
102   // inject functionality
103   inject(this, new X.loadable()); // this object is loadable from a file
104   inject(this, new X.thresholdable()); // this object is thresholdable
105   
106 };
107 // inherit from X.base
108 goog.inherits(X.scalars, X.base);
109

111 /**
112  * Get the array containing the scalars. This array is 'WebGL-ready', meaning
113  * that all values are represented by 3 elements to match the X,Y,Z coordinates
114  * of points.
115  * 
116  * @public
117  */
118 X.scalars.prototype.__defineGetter__('array', function() {
119 
120   return this._array;
121   
122 });
123 
124 
125 /**
126  * Set the array containing the scalars. This array has to be WebGL-ready
127  * meaning that it has to match X.object.points.length, which equals 3 entries
128  * for each vertex (X,Y,Z coordinates). Calling this method marks this object as
129  * dirty so the X.renderer can pick it up. This method should be used
130  * externally.
131  * 
132  * @param {Array} array The WebGL-ready array matching X.object._points.length
133  *          in size.
134  * @public
135  */
136 X.scalars.prototype.__defineSetter__('array', function(array) {
137 
138   this._array = array;
139   this._glArray = array;
140   
141   // also, mark this dirty so the renderer can pick it up
142   this._dirty = true;
143   
144 });
145

147 /**
148  * Get the min color which is used to map the scalars to colors in a linear
149  * fashion.
150  * 
151  * @return {!Array} An array holding the r,g,b components of the color.
152  * @public
153  */
154 X.scalars.prototype.__defineGetter__('minColor', function() {
155 
156   return this._minColor;
157   
158 });
159 
160 
161 /**
162  * Set the min color to linear map the scalars to colors.
163  * 
164  * @param {!Array} minColor The color corresponding to the min. scalar value.
165  * @public
166  */
167 X.scalars.prototype.__defineSetter__('minColor', function(minColor) {
168 
169   if (!goog.isDefAndNotNull(minColor) || !(minColor instanceof Array) ||
170       (minColor.length != 3)) {
171     
172     throw new Error('Invalid min. color.');
173     
174   }
175   
176   this._minColor = minColor;
177   
178 });
179 
180
