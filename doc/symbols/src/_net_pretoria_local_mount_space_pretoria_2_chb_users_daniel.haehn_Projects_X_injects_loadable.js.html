 30 // provides
 31 goog.provide('X.loadable');
 32 
 33 // requires
 34 goog.require('X.base');
 35 goog.require('X.file');
 36 
 37 
 38 
 39 /**
 40  * Injective mix-in for all loadable objects.
 41  * 
 42  * @constructor
 43  */
 44 X.loadable = function() {
 45 
 46   /**
 47    * The file of this loadable object.
 48    * 
 49    * @type {?X.file}
 50    * @protected
 51    */
 52   this._file = null;
 53   
 54 };
 55 
 56 
 57 /**
 58  * Load this object from a file path or reset the associated file path.
 59  * 
 60  * @param {?string} filepath The file path/URL to load. If null, reset the
 61  *          associated file.
 62  */
 63 X.loadable.prototype.__defineSetter__('file', function(filepath) {
 64 
 65   if (!goog.isDefAndNotNull(filepath)) {
 66     
 67     // if path is null, we reset the associated X.file object
 68     
 69     this._file = null;
 70     return;
 71     
 72   }
 73   
 74   this._file = new X.file(filepath);
 75   
 76 });
 77 
 78 
 79 /**
 80  * Get the associated X.file for this object.
 81  * 
 82  * @return {string} The associated X.file or null if no file is associated.
 83  */
 84 X.loadable.prototype.__defineGetter__('file', function() {
 85 
 86   if (!this._file) {
 87     
 88     return '';
 89     
 90   }
 91   
 92   return this._file._path;
 93   
 94 });
