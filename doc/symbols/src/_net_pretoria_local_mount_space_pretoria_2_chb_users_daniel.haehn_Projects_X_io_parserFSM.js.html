 45 /**
 46  * Create a parser for the binary freesurfer meshes
 47  * 
 48  * @constructor
 49  * @extends X.parser
 50  */
 51 X.parserFSM = function() {
 52 
 53   //
 54   // call the standard constructor of X.parser
 55   goog.base(this);
 56   
 57   //
 58   // class attributes
 59   
 60   /**
 61    * @inheritDoc
 62    * @const
 63    */
 64   this._classname = 'parserFSM';
 65   
 66 };
 67 // inherit from X.parser
 68 goog.inherits(X.parserFSM, X.parser);
 69 
 70 
 71 /**
 72  * @inheritDoc
 73  */
 74 X.parserFSM.prototype.parse = function(container, object, data, flag) {
 75 
 76   var p = object._points;
 77   var n = object._normals;
 78   var ind = object._pointIndices;
 79   
 80   // in .FSM files, the points are not ordered for rendering, so we need to
 81   // buffer everything in X.triplets containers and then order it
 82   var unorderedPoints = new X.triplets();
 83   
 84   var currentOffset = 0;
 85   
 86   // parse some magic bytes
 87   this.parseString(data, currentOffset, 3);
 88   currentOffset += 3;
 89   
 90   // Go through two newlines
 91   var iters = 0;
 92   var curChar;
 93   do {
 94     curChar = this.parseUChar8(data, currentOffset++);
 95     iters++;
 96   } while ((iters < 200) && (curChar != 0x0A))
 97 
 98   // Read one more newline
 99   curChar = this.parseUChar8(data, currentOffset++);
100   
101   // get the number of vertices
102   var numberOfVertices = this.parseUInt32EndianSwapped(data, currentOffset);
103   currentOffset += 4;
104   
105   // get the number of triangles
106   var numberOfTriangles = this.parseUInt32EndianSwapped(data, currentOffset);
107   currentOffset += 4;
108   
109   // we count the appearance of indices to be able to average the normals
110   var indexCounter = [];
111   
112   // parse the coordinates from the data
113   var v = 0;
114   for (v = 0; v < numberOfVertices; v++) {
115     
116     // grab 3 floats which are X,Y,Z in world coordinates
117     var x = this.parseFloat32EndianSwapped(data, currentOffset);
118     currentOffset += 4;
119     var y = this.parseFloat32EndianSwapped(data, currentOffset);
120     currentOffset += 4;
121     var z = this.parseFloat32EndianSwapped(data, currentOffset);
122     currentOffset += 4;
123     
124     // initialize the index counter with 0
125     indexCounter[v] = 0;
126     
127     // store the points unordered
128     unorderedPoints.add(x, y, z);
129     
130   }
131   
132   // a list of triangles composed by indices
133   var triangles = [];
134   
135   // a list of normals stored by index
136   var normals = [];
137   
138   // parse the indices from the data
139   var t;
140   for (t = 0; t < numberOfTriangles; t++) {
141     
142     // grab the three indices which define a triangle
143     var index1 = this.parseUInt32EndianSwapped(data, currentOffset);
144     currentOffset += 4;
145     var index2 = this.parseUInt32EndianSwapped(data, currentOffset);
146     currentOffset += 4;
147     var index3 = this.parseUInt32EndianSwapped(data, currentOffset);
148     currentOffset += 4;
149     
150     // we count the appearance of the indices
151     indexCounter[index1] += 1;
152     indexCounter[index2] += 1;
153     indexCounter[index3] += 1;
154     
155     triangles.push([index1, index2, index3]);
156     normals.push(new goog.math.Vec3(0, 0, 0));
157   }
158   
159   // compute the triangle normals using the indices
160   for (t = 0; t < numberOfTriangles; t++) {
161     
162     var index1 = triangles[t][0];
163     var index2 = triangles[t][1];
164     var index3 = triangles[t][2];
165     
166     // create a triangle with the 3 vertices
167     var v1 = unorderedPoints.get(index1);
168     var v2 = unorderedPoints.get(index2);
169     var v3 = unorderedPoints.get(index3);
170     
171     //
172     // compute the normals
173     var v1v = new goog.math.Vec3(v1[0], v1[1], v1[2]);
174     var v2v = new goog.math.Vec3(v2[0], v2[1], v2[2]);
175     var v3v = new goog.math.Vec3(v3[0], v3[1], v3[2]);
176     
177     var n1 = v2v.clone().subtract(v1v);
178     var n2 = v3v.clone().subtract(v1v);
179     
180     var normal = goog.math.Vec3.cross(n1, n2).normalize();
181     
182     normals[index1] = normals[index1].add(normal);
183     normals[index2] = normals[index2].add(normal);
184     normals[index3] = normals[index3].add(normal);
185     
186   }
187   
188   // configure the points and normals for the X.object
189   for (t = 0; t < numberOfTriangles; t++) {
190     
191     var index1 = triangles[t][0];
192     var index2 = triangles[t][1];
193     var index3 = triangles[t][2];
194     
195     // create a triangle with the 3 vertices
196     var v1 = unorderedPoints.get(index1);
197     var v2 = unorderedPoints.get(index2);
198     var v3 = unorderedPoints.get(index3);
199     
200     // store the ordered vertex indices
201     ind.push(index1);
202     ind.push(index2);
203     ind.push(index3);
204     
205     // store the ordered vertices
206     p.add(v1[0], v1[1], v1[2]);
207     p.add(v2[0], v2[1], v2[2]);
208     p.add(v3[0], v3[1], v3[2]);
209     
210     // transform triangle normals to vertex normals
211     var normal1 = normals[index1].scale(1 / indexCounter[index1]).normalize();
212     var normal2 = normals[index2].scale(1 / indexCounter[index2]).normalize();
213     var normal3 = normals[index3].scale(1 / indexCounter[index3]).normalize();
214     
215     // .. add'em
216     n.add(normal1.x, normal1.y, normal1.z);
217     n.add(normal2.x, normal2.y, normal2.z);
218     n.add(normal3.x, normal3.y, normal3.z);
219     
220   }
221   
222   // .. and set the objectType to triangles
223   object._type = X.displayable.types.TRIANGLES;
224   
225   // the object should be set up here, so let's fire a modified event
226   var modifiedEvent = new X.event.ModifiedEvent();
227   modifiedEvent._object = object;
228   modifiedEvent._container = container;
229   this.dispatchEvent(modifiedEvent);
230   
231 };
