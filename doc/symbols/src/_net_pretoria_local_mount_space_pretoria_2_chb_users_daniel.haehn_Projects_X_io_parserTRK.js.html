 44 /**
 45  * Create a parser for the binary .TRK format.
 46  * 
 47  * @constructor
 48  * @extends X.parser
 49  */
 50 X.parserTRK = function() {
 51 
 52   //
 53   // call the standard constructor of X.base
 54   goog.base(this);
 55   
 56   //
 57   // class attributes
 58   
 59   /**
 60    * @inheritDoc
 61    * @const
 62    */
 63   this._classname = 'parserTRK';
 64   
 65 };
 66 // inherit from X.parser
 67 goog.inherits(X.parserTRK, X.parser);
 68 
 69 
 70 /**
 71  * @inheritDoc
 72  */
 73 X.parserTRK.prototype.parse = function(container, object, data, flag) {
 74 
 75   var p = object._points;
 76   var n = object._normals;
 77   var c = object._colors;
 78   
 79   var offset = 0;
 80   
 81   // parse the header of the .TRK file
 82   // Documented here: http://trackvis.org/docs/?subsect=fileformat
 83   var header = {
 84     'id_string': this.parseString(data, offset, 6),
 85     'dim': this.parseUInt16Array(data, offset + 6, 3)[0],
 86     'voxel_size': this.parseFloat32Array(data, offset + 12, 3)[0],
 87     'origin': this.parseFloat32Array(data, offset + 24, 3)[0],
 88     'n_scalars': this.parseUInt16(data, offset + 36),
 89     'scalar_name': this.parseString(data, offset + 38, 200),
 90     'n_properties': this.parseUInt16(data, offset + 238),
 91     'property_name': this.parseString(data, offset + 240, 200),
 92     'vox_to_ras': this.parseFloat32Array(data, offset + 440, 16)[0],
 93     'reserved': this.parseString(data, offset + 504, 444),
 94     'voxel_order': this.parseString(data, offset + 948, 4),
 95     'pad2': this.parseString(data, offset + 952, 4),
 96     'image_orientation_patient': this.parseFloat32Array(data, offset + 956, 6)[0],
 97     'pad1': this.parseString(data, offset + 980, 2),
 98     'invert_x': this.parseUChar8(data, offset + 982),
 99     'invert_y': this.parseUChar8(data, offset + 983),
100     'invert_z': this.parseUChar8(data, offset + 984),
101     'swap_xy': this.parseUChar8(data, offset + 985),
102     'swap_yz': this.parseUChar8(data, offset + 986),
103     'swap_zx': this.parseUChar8(data, offset + 987),
104     'n_count': this.parseUInt32(data, offset + 988),
105     'version': this.parseUInt32(data, offset + 992),
106     'hdr_size': this.parseUInt32(data, offset + 996)
107   };
108   
109   //
110   // parse the data
111   offset = header.hdr_size;
112   
113   var numberOfFibers = header.n_count;
114   
115   // loop through all fibers
116   var fibers = [];
117   var lengths = [];
118   var minLength = Infinity;
119   var maxLength = -Infinity;
120   
121   var minX = null;
122   var maxX = null;
123   var minY = null;
124   var maxY = null;
125   var minZ = null;
126   var maxZ = null;
127   
128   var i;
129   for (i = 0; i < numberOfFibers; i++) {
130     var numPoints = this.parseUInt32(data, offset);
131     
132     var currentPoints = new X.triplets();
133     
134     var length = 0.0;
135     
136     offset += 4;
137     
138     // loop through the points of this fiber
139     for ( var j = 0; j < numPoints; j++) {
140       
141       // read coordinates
142       var x = this.parseFloat32(data, offset);
143       offset += 4;
144       
145       var y = this.parseFloat32(data, offset);
146       offset += 4;
147       
148       var z = this.parseFloat32(data, offset);
149       offset += 4;
150       
151       // read scalars
152       var scalars = null;
153       if (header.n_scalars > 0) {
154         scalars = this.parseFloat32Array(data, offset, header.n_scalars)[0];
155         offset += (header.n_scalars * 4);
156       }
157       
158       // Convert coordinates to world space by dividing by spacing
159       x = x / header.voxel_size[0];
160       y = y / header.voxel_size[1];
161       z = z / header.voxel_size[2];
162       
163       currentPoints.add(x, y, z);
164       
165       // fiber length
166       if (j > 0) {
167         // if not the first point, calculate length
168         var oldPoint = currentPoints.get(j - 1);
169         
170         length += Math.sqrt(Math.pow(x - oldPoint[0], 2) +
171             Math.pow(y - oldPoint[1], 2) + Math.pow(z - oldPoint[2], 2));
172         
173 
174       }
175       
176     }
177     
178     // read additional properties, if existing
179     // we don't support them right now in XTK
180     if (header.n_properties > 0) {
181       // var properties = this
182       // .parseFloat32Array(data, offset, header.n_properties);
183       offset += (header.n_properties * 4);
184     }
185     
186 
187     // we need to get the bounding box of the whole .trk file before we add the
188     // points to properly setup normals
189     
190     var cMinX = currentPoints._minA;
191     var cMaxX = currentPoints._maxA;
192     var cMinY = currentPoints._minB;
193     var cMaxY = currentPoints._maxB;
194     var cMinZ = currentPoints._minC;
195     var cMaxZ = currentPoints._maxC;
196     
197     if (!minX || cMinX < minX) {
198       minX = cMinX;
199     }
200     if (!maxX || cMaxX > maxX) {
201       maxX = cMaxX;
202     }
203     if (!minY || cMinY < minY) {
204       minY = cMinY;
205     }
206     if (!maxY || cMaxY > maxY) {
207       maxY = cMaxY;
208     }
209     if (!minZ || cMinZ < minZ) {
210       minZ = cMinZ;
211     }
212     if (!maxZ || cMaxZ > maxZ) {
213       maxZ = cMaxZ;
214     }
215     
216     // append this track to our fibers list
217     fibers.push(currentPoints);
218     // .. and also the length
219     lengths.push(length);
220     
221   } // end of loop through all tracks
222   
223   // calculate the center based on the bounding box of the whole .trk file
224   var centerX = (minX + maxX) / 2;
225   var centerY = (minY + maxY) / 2;
226   var centerZ = (minZ + maxZ) / 2;
227   
228   // the scalar array
229   var scalarArray = [];
230   
231   // now we have a list of fibers
232   for (i = 0; i < numberOfFibers; i++) {
233     
234     // grab the current points of this fiber
235     var points = fibers[i];
236     var numberOfPoints = points.count;
237     // grab the length of this fiber
238     var length = lengths[i];
239     
240     minLength = Math.min(minLength, length);
241     maxLength = Math.max(maxLength, length);
242     
243     for ( var j = 0; j < numberOfPoints - 1; j++) {
244       
245       // TODO min max check?
246       
247       // grab the point with the currentIndex
248       var currentPoint = points.get(j);
249       // and connect it to the next one
250       var nextPoint = points.get(j + 1);
251       
252       // .. and add both
253       p.add(currentPoint[0], currentPoint[1], currentPoint[2]);
254       p.add(nextPoint[0], nextPoint[1], nextPoint[2]);
255       
256       // calculate and add the normals for the two points
257       var nCurrentPointX = currentPoint[0] - centerX;
258       var nCurrentPointY = currentPoint[1] - centerY;
259       var nCurrentPointZ = currentPoint[2] - centerZ;
260       var nCurrentPointLength = Math.sqrt(nCurrentPointX * nCurrentPointX +
261           nCurrentPointY * nCurrentPointY + nCurrentPointZ * nCurrentPointZ);
262       var nNextPointX = nextPoint[0] - centerX;
263       var nNextPointY = nextPoint[1] - centerY;
264       var nNextPointZ = nextPoint[2] - centerZ;
265       var nNextPointLength = Math.sqrt(nNextPointX * nNextPointX + nNextPointY *
266           nNextPointY + nNextPointZ * nNextPointZ);
267       
268       n.add(nCurrentPointX / nCurrentPointLength, nCurrentPointY /
269           nCurrentPointLength, nCurrentPointZ / nCurrentPointLength);
270       n.add(nNextPointX / nNextPointLength, nNextPointY / nNextPointLength,
271           nNextPointZ / nNextPointLength);
272       
273       var start = currentPoint;
274       var end = nextPoint;
275       
276       var diff = [Math.abs(end[0] - start[0]), Math.abs(end[1] - start[1]),
277                   Math.abs(end[2] - start[2])];
278       
279       var distance = Math.sqrt(diff[0] * diff[0] + diff[1] * diff[1] + diff[2] *
280           diff[2]);
281       
282       diff[0] /= distance;
283       diff[1] /= distance;
284       diff[2] /= distance;
285       
286       // add the color
287       c.add(diff[0], diff[1], diff[2]);
288       c.add(diff[0], diff[1], diff[2]);
289       
290       // add the length (6 times since we added two points with each 3
291       // coordinates)
292       scalarArray.push(length);
293       scalarArray.push(length);
294       scalarArray.push(length);
295       scalarArray.push(length);
296       scalarArray.push(length);
297       scalarArray.push(length);
298       
299     } // loop through points
300     
301   } // loop through fibers
302   
303   // set the object type to LINES
304   object._type = X.displayable.types.LINES;
305   
306   // attach the scalars
307   var scalars = new X.scalars();
308   scalars._min = minLength;
309   scalars._max = maxLength;
310   scalars._lowerThreshold = minLength;
311   scalars._upperThreshold = maxLength;
312   scalars._glArray = scalarArray; // the ordered, gl-Ready
313   // version - use the setter to mark the scalars dirty
314   scalars._replaceMode = false; // we don't want to replace - we want to
315   // discard!
316   scalars._dirty = true;
317   object._scalars = scalars;
318   
319   // the object should be set up here, so let's fire a modified event
320   var modifiedEvent = new X.event.ModifiedEvent();
321   modifiedEvent._object = object;
322   modifiedEvent._container = container;
323   this.dispatchEvent(modifiedEvent);
324   
325 };
