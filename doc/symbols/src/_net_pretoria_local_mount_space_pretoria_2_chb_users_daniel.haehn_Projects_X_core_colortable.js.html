 30 // provides
 31 goog.provide('X.colortable');
 32 
 33 // requires
 34 goog.require('X.base');
 35 goog.require('X.loadable');
 36 goog.require('goog.structs.Map');
 37 
 38 
 39 
 40 /**
 41  * Create a container for a colortable.
 42  * 
 43  * @constructor
 44  * @extends X.base
 45  */
 46 X.colortable = function() {
 47 
 48   //
 49   // call the standard constructor of X.base
 50   goog.base(this);
 51   
 52   //
 53   // class attributes
 54   
 55   /**
 56    * @inheritDoc
 57    * @const
 58    */
 59   this._classname = 'colortable';
 60   
 61   /**
 62    * The internal hash map to store the value-color mapping.
 63    * 
 64    * @type {!goog.structs.Map}
 65    * @protected
 66    */
 67   this._map = new goog.structs.Map();
 68   
 69   // inject functionality
 70   inject(this, new X.loadable()); // this object is loadable from a file
 71   
 72 };
 73 // inherit from X.base
 74 goog.inherits(X.colortable, X.base);
 75 
 76 
 77 /**
 78  * Add an entry to this color table.
 79  * 
 80  * @param {!number} value The number to map to a color.
 81  * @param {!string} label The label.
 82  * @param {!number} r The red component.
 83  * @param {!number} g The green component.
 84  * @param {!number} b The blue component.
 85  * @param {!number} a The alpha component.
 86  * @throws {Error} An error, if the given values are invalid.
 87  */
 88 X.colortable.prototype.add = function(value, label, r, g, b, a) {
 89 
 90   if (!goog.isNumber(value) || !goog.isNumber(r) || !goog.isNumber(g) ||
 91       !goog.isNumber(b) || !goog.isNumber(a)) {
 92     
 93     throw new Error('Invalid color table entry.');
 94     
 95   }
 96   
 97   this._map.set(value, [label, r, g, b, a]);
 98   
 99   this._dirty = true;
100   
101 };
