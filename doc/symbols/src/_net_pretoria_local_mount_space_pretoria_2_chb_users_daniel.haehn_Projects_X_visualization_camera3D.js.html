 39 /**
 40  * Create a 3D perspective camera.
 41  * 
 42  * @constructor
 43  * @param {number} width The width of the camera's viewport.
 44  * @param {number} height The height of the camera's viewport.
 45  * @extends X.camera
 46  */
 47 X.camera3D = function(width, height) {
 48 
 49   //
 50   // call the standard constructor of X.base
 51   goog.base(this, width, height);
 52   
 53   //
 54   // class attributes
 55   
 56   /**
 57    * @inheritDoc
 58    * @const
 59    */
 60   this._classname = 'camera3D';
 61   
 62   /**
 63    * The field of view in degrees.
 64    * 
 65    * @type {!number}
 66    * @const
 67    */
 68   this._fieldOfView = 45;
 69   
 70   /**
 71    * The perspective matrix.
 72    * 
 73    * @type {!Object}
 74    * @protected
 75    */
 76   this._perspective = new Float32Array(this.calculatePerspective_(
 77       this._fieldOfView, (width / height), 1, 10000).flatten());
 78   
 79 };
 80 // inherit from X.base
 81 goog.inherits(X.camera3D, X.camera);

 84 /**
 85  * Observe ROTATE events for this 3D camera.
 86  * 
 87  * @inheritDoc
 88  */
 89 X.camera3D.prototype.observe = function(interactor) {
 90 
 91   goog.base(this, 'observe', interactor);
 92   
 93   goog.events.listen(interactor, X.event.events.ROTATE, this.onRotate_
 94       .bind(this));
 95   
 96 };

 99 /**
100  * The callback for a ROTATE event.
101  * 
102  * @param {!X.event.RotateEvent} event The event.
103  * @throws {Error} An exception if the event is invalid.
104  * @protected
105  */
106 X.camera3D.prototype.onRotate_ = function(event) {
107 
108   if (!(event instanceof X.event.RotateEvent)) {
109     
110     throw new Error('Received no valid rotate event.');
111     
112   }
113   
114   this.rotate(event._distance);
115   
116 };

119 /**
120  * Calculate a perspective matrix based on the given values. This calculation is
121  * based on known principles of Computer Vision (Source: TODO?).
122  * 
123  * @param {number} fieldOfViewY The field of view in degrees in Y direction.
124  * @param {number} aspectRatio The aspect ratio between width and height of the
125  *          viewport.
126  * @param {number} zNearClippingPlane The Z coordinate of the near clipping
127  *          plane (close to the eye).
128  * @param {number} zFarClippingPlane The Z coordinate of the far clipping plane
129  *          (far from the eye).
130  * @return {X.matrix} The perspective matrix.
131  * @private
132  */
133 X.camera3D.prototype.calculatePerspective_ = function(fieldOfViewY,
134     aspectRatio, zNearClippingPlane, zFarClippingPlane) {
135 
136   var ymax = zNearClippingPlane * Math.tan(fieldOfViewY * Math.PI / 360.0);
137   var ymin = -ymax;
138   var xmin = ymin * aspectRatio;
139   var xmax = ymax * aspectRatio;
140   
141   return this.calculateViewingFrustum_(xmin, xmax, ymin, ymax,
142       zNearClippingPlane, zFarClippingPlane);
143   
144 };

147 /**
148  * Calculate the viewing frustum which is the three-dimensional area which is
149  * visible to the eye by 'trimming' the world space. This calculation is based
150  * on known principles of Computer Vision (Source:
151  * http://en.wikipedia.org/wiki/Viewing_frustum).
152  * 
153  * @param {number} left The Y coordinate of the left border.
154  * @param {number} right The Y coordinate of the right border.
155  * @param {number} bottom The X coordinate of the bottom border.
156  * @param {number} top The X coordinate of the top border.
157  * @param {number} znear The Z coordinate of the near the eye border.
158  * @param {number} zfar The Z coordinate of the far of the eye border.
159  * @return {X.matrix} The frustum matrix.
160  * @private
161  */
162 X.camera3D.prototype.calculateViewingFrustum_ = function(left, right, bottom,
163     top, znear, zfar) {
164 
165   var x = 2 * znear / (right - left);
166   var y = 2 * znear / (top - bottom);
167   var a = (right + left) / (right - left);
168   var b = (top + bottom) / (top - bottom);
169   var c = -(zfar + znear) / (zfar - znear);
170   var d = -2 * zfar * znear / (zfar - znear);
171   
172   return new X.matrix([[x, 0, a, 0], [0, y, b, 0], [0, 0, c, d], [0, 0, -1, 0]]);
173   
174 };

177 /**
178  * Perform a rotate operation. This method fires a X.event.RenderEvent() after
179  * the calculation is done.
180  * 
181  * @param {!goog.math.Vec2|!Array} distance The distance of the rotation in
182  *          respect of the last camera position as either a 2D Array or a
183  *          goog.math.Vec2 containing the X and Y distances for the rotation.
184  * @public
185  */
186 X.camera3D.prototype.rotate = function(distance) {
187 
188   if ((distance instanceof Array) && (distance.length == 2)) {
189     
190     distance = new goog.math.Vec2(distance[0], distance[1]);
191     
192   } else if (!(distance instanceof goog.math.Vec2)) {
193     
194     throw new Error('Invalid distance vector for rotate operation.');
195     
196   }
197   
198   // in radii, the 5 is a constant stating how quick the rotation performs..
199   var angleX = -distance.x / 5 * Math.PI / 180;
200   var angleY = -distance.y / 5 * Math.PI / 180;
201   
202   var identity = X.matrix.createIdentityMatrix(4);
203   // the x-Axis vector is determined by the first row of the view matrix
204   var xAxisVector = new goog.math.Vec3(parseFloat(this.view.getValueAt(0, 0)),
205       parseFloat(this.view.getValueAt(0, 1)), parseFloat(this.view.getValueAt(
206           0, 2)));
207   // the y-Axis vector is determined by the second row of the view matrix
208   var yAxisVector = new goog.math.Vec3(parseFloat(this.view.getValueAt(1, 0)),
209       parseFloat(this.view.getValueAt(1, 1)), parseFloat(this.view.getValueAt(
210           1, 2)));
211   
212   // we rotate around the Y Axis when the mouse moves along the screen in X
213   // direction
214   var rotateX = identity.rotate(angleX, yAxisVector);
215   
216   // we rotate around the X axis when the mouse moves along the screen in Y
217   // direction
218   var rotateY = identity.rotate(angleY, xAxisVector);
219   
220   // perform the actual rotation calculation
221   this.view = new X.matrix(this.view.multiply(rotateY.multiply(rotateX)));
222   this._glview = new Float32Array(this.view.flatten());
223   
224   // fire a render event
225   this.dispatchEvent(new X.event.RenderEvent());
226   
227 };

230 /**
231  * @inheritDoc
232  */
233 X.camera3D.prototype.lookAt_ = function(cameraPosition, targetPoint) {
234 
235   goog.base(this, 'lookAt_', cameraPosition, targetPoint);
236   
237   var matrix = X.matrix.createIdentityMatrix(4);
238   
239   // Make rotation matrix
240   
241   // Z vector = cameraPosition - targetPoint
242   var zVector = goog.math.Vec3.difference(cameraPosition, targetPoint);
243   
244   // normalize Z
245   zVector = zVector.normalize();
246   
247   // Y vector = up
248   var yVector = this._up.clone();
249   
250   // WARNING: there is a problem if yVector == zVector
251   if (yVector.equals(zVector)) {
252     
253     // we now change the zVector a little bit
254     yVector.z += 0.000001;
255     
256   }
257   
258   // X vector = Y x Z
259   var xVector = goog.math.Vec3.cross(yVector, zVector);
260   
261   // recompute Y vector = Z x X
262   yVector = goog.math.Vec3.cross(zVector, xVector);
263   
264   // normalize X and Y
265   xVector = xVector.normalize();
266   yVector = yVector.normalize();
267   
268   // create view matrix
269   // first row
270   matrix.setValueAt(0, 0, xVector.x);
271   matrix.setValueAt(0, 1, xVector.y);
272   matrix.setValueAt(0, 2, xVector.z);
273   matrix.setValueAt(0, 3, 0);
274   
275   // second row
276   matrix.setValueAt(1, 0, yVector.x);
277   matrix.setValueAt(1, 1, yVector.y);
278   matrix.setValueAt(1, 2, yVector.z);
279   matrix.setValueAt(1, 3, 0);
280   
281   // third row
282   matrix.setValueAt(2, 0, zVector.x);
283   matrix.setValueAt(2, 1, zVector.y);
284   matrix.setValueAt(2, 2, zVector.z);
285   matrix.setValueAt(2, 3, 0);
286   
287   // last row
288   matrix.setValueAt(3, 0, 0);
289   matrix.setValueAt(3, 1, 0);
290   matrix.setValueAt(3, 2, 0);
291   matrix.setValueAt(3, 3, 1);
292   
293   var invertedCameraPosition = cameraPosition.clone();
294   
295   return matrix.translate(invertedCameraPosition.invert());
296   
297 };
