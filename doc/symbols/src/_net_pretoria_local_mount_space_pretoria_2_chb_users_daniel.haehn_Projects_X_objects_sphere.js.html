 30 // provides
 31 goog.provide('X.sphere');

 33 // requires
 34 goog.require('CSG.sphere');
 35 goog.require('X.base');
 36 goog.require('X.constructable');
 37 goog.require('X.object');

 41 /**
 42  * Create a displayable Sphere.
 43  * 
 44  * @constructor
 45  * @extends X.object
 46  */
 47 X.sphere = function() {
 48 
 49   //
 50   // call the standard constructor of X.base
 51   goog.base(this);
 52   
 53   //
 54   // class attributes
 55   
 56   /**
 57    * @inheritDoc
 58    * @const
 59    */
 60   this._classname = 'sphere';
 61   
 62   /**
 63    * The center of this sphere in the world space.
 64    * 
 65    * @type {!Array}
 66    * @protected
 67    */
 68   this._center = [0, 0, 0];
 69   
 70   /**
 71    * The radius of this sphere.
 72    * 
 73    * @type {!number}
 74    * @protected
 75    */
 76   this._radius = 5;
 77   
 78   /**
 79    * The number of slices to generate the sphere shape.
 80    * 
 81    * @type {!number}
 82    * @protected
 83    */
 84   this._slices = 32;
 85   
 86   /**
 87    * The number of stacks to generate the sphere shape.
 88    * 
 89    * @type {!number}
 90    * @protected
 91    */
 92   this._stacks = 16;
 93   
 94   inject(this, new X.constructable()); // this object is constructable
 95   
 96 };
 97 // inherit from X.object
 98 goog.inherits(X.sphere, X.object);
 99 
100 
101 /**
102  * Get the center of this sphere.
103  * 
104  * @return {!Array} The center as an array with length 3.
105  * @public
106  */
107 X.sphere.prototype.__defineGetter__('center', function() {
108 
109   return this._center;
110   
111 });
112 
113 
114 /**
115  * Set the center of this sphere.
116  * 
117  * @param {!Array} center The center as an array with length 3 ([X,Y,Z]).
118  * @throws {Error} An error, if the center is invalid.
119  * @public
120  */
121 X.sphere.prototype.__defineSetter__('center', function(center) {
122 
123   if (!goog.isDefAndNotNull(center) || !(center instanceof Array) ||
124       (center.length != 3)) {
125     
126     throw new Error('Invalid center');
127     
128   }
129   
130   this._center = center;
131   
132 });
133 
134 
135 /**
136  * Get the radius of this sphere.
137  * 
138  * @return {!number} The radius.
139  * @public
140  */
141 X.sphere.prototype.__defineGetter__('radius', function() {
142 
143   return this._radius;
144   
145 });
146 
147 
148 /**
149  * Set the radius of this sphere.
150  * 
151  * @param {!number} radius The radius.
152  * @throws {Error} An error, if the given radius is invalid.
153  * @public
154  */
155 X.sphere.prototype.__defineSetter__('radius', function(radius) {
156 
157   if (!goog.isNumber(radius)) {
158     
159     throw new Error('Invalid radius.');
160     
161   }
162   
163   this._radius = radius;
164   
165 });
166 
167 
168 /**
169  * @inheritDoc
170  */
171 X.sphere.prototype.modified = function() {
172 
173   this.fromCSG(new CSG.sphere({
174     center: this._center,
175     radius: this._radius,
176     slices: this._slices,
177     stacks: this._stacks
178   }));
179   
180   // call the super class
181   goog.base(this, 'modified');
182   
183 };

185 // export symbols (required for advanced compilation)
186 goog.exportSymbol('X.sphere', X.sphere);
187 goog.exportSymbol('X.sphere.prototype.modified', X.sphere.prototype.modified);
