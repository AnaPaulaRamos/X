 48 /**
 49  * This object loads external files in an asynchronous fashion. In addition, the
 50  * loading process is monitored and summarized to a total progress value.
 51  * 
 52  * @constructor
 53  * @extends X.base
 54  */
 55 X.loader = function() {
 56 
 57   // call the standard constructor of X.base
 58   goog.base(this);
 59   
 60   //
 61   // class attributes
 62   
 63   /**
 64    * @inheritDoc
 65    * @const
 66    */
 67   this._classname = 'loader';
 68   
 69   /**
 70    * The hash map holding all the jobs and their status.
 71    * 
 72    * @type {!goog.structs.Map}
 73    * @protected
 74    */
 75   this._jobs = new goog.structs.Map();
 76   
 77   /**
 78    * The overall progress value in the range of 0 and 1.
 79    * 
 80    * @type {!number}
 81    * @protected
 82    */
 83   this._progress = 0;
 84   
 85 };
 86 // inherit from X.base
 87 goog.inherits(X.loader, X.base);

 90 /**
 91  * Check if all loading is completed.
 92  * 
 93  * @return {boolean} TRUE if all loading is completed, FALSE else wise.
 94  */
 95 X.loader.prototype.completed = function() {
 96 
 97   // now we check if all of our jobs are completed
 98   return !this._jobs.containsValue(false);
 99   
100 };

103 /**
104  * Add a value to the overall progress value. The given value is normalized
105  * according to the number of current jobs. After adding the value, a
106  * X.event.ProgressEvent is fired to inform the progress bar.
107  * 
108  * @param {!number} value The value to add.
109  */
110 X.loader.prototype.addProgress = function(value) {
111 
112   // we have a three stage system during loading
113   //
114   // stage 1: loading in X.loader
115   // stage 2: parsing in X.parser (when loading textures, this does not happen
116   // so the loading adds a 2 in total)
117   // stage 3: setting up in X.renderer
118   //
119   // each stage adds progress from 0..1 with a total of 1 at the end
120   // add a fake job to prevent the user starring at a full progress bar
121   this._progress += value / (this._jobs.getCount()) / 3;
122   
123   // clamp progress to 0..1
124   this._progress = Math.min(1, this._progress);
125   
126   // fire a progress event so the progressbar can display something
127   var progressEvent = new X.event.ProgressEvent();
128   progressEvent._value = this._progress;
129   this.dispatchEvent(progressEvent);
130   
131 };

134 /**
135  * Check if the given container is configured with a valid file format.
136  * 
137  * @param {!X.base} container A container which has an X.file() attached.
138  * @return {!Array} An array holding the following information [filepath,
139  *         extension in uppercase without the '.', the associated but
140  *         uninstantiated parser for this file format, additional (optional)
141  *         flags which are passed to the parser, the response type]
142  * @throws {Error} An error, if the configured file format is not supported.
143  */
144 X.loader.prototype.checkFileFormat = function(container) {
145 
146   // get the associated file of the object
147   var filepath = container._file._path;
148   
149   // grab the file extension
150   var extension = filepath.split('.').pop().toUpperCase();
151   
152   // check if the file format is supported
153   if (!(extension in X.loader.extensions)) {
154     
155     throw new Error('The ' + extension + ' file format is not supported.');
156     
157   }
158   
159   return [filepath, extension, X.loader.extensions[extension][0],
160           X.loader.extensions[extension][1], X.loader.extensions[extension][2]];
161   
162 };

165 /**
166  * Download a file associated to a container via Ajax. Callbacks are executed
167  * depending on the success of the request.
168  * 
169  * @param {!X.base} container The container which has a X.file() attached.
170  * @param {!X.object} object The X.object which is the parent of the container
171  *          or equals it.
172  * @throws {Error} An error, if the given objects were invalid.
173  */
174 X.loader.prototype.load = function(container, object) {
175 
176   if (!container || !object) {
177     
178     // should not happen
179     throw new Error('No container or object to load.');
180     
181   }
182   
183   // check the file format which returns the filepath, extension and the parser
184   var _checkresult = this.checkFileFormat(container);
185   var filepath = _checkresult[0];
186   var responseType = _checkresult[4];
187   
188   // we use a simple XHR to get the file contents
189   // this works for binary and for ascii files
190   var request = new XMLHttpRequest();
191   
192   // listen to abort events
193   goog.events.listen(request, 'abort', this.failed.bind(this, request,
194       container, object));
195   
196   // listen to error events
197   goog.events.listen(request, 'error', this.failed.bind(this, request,
198       container, object));
199   
200   // listen to completed events which triggers parsing
201   goog.events.listen(request, 'load', this.parse.bind(this, request, container,
202       object));
203   
204   // configure the URL
205   request.open('GET', filepath, true);
206   if (responseType) {
207     // set the response type if != null, else fall back to the default 'text'
208     request.responseType = responseType;
209   }
210   request.overrideMimeType("text/plain; charset=x-user-defined");
211   request.setRequestHeader("Content-Type", "text/plain");
212   
213   // .. and GO!
214   request.send(null);
215   
216   // add this loading job to our jobs map
217   this._jobs.set(container._id, false);
218   
219 };

222 /**
223  * Trigger parsing of a data stream. This is the callback for successful
224  * downloading. A single-shot listener is configured for a X.event.ModifiedEvent
225  * which then triggers the complete callback.
226  * 
227  * @param {!XMLHttpRequest} request The original XHR.
228  * @param {!X.base} container The container which has a X.file() attached.
229  * @param {!X.object} object The X.object which is the parent of the container
230  *          or equals it.
231  */
232 X.loader.prototype.parse = function(request, container, object) {
233 
234   // downloading completed, add progress
235   this.addProgress(1.0);
236   
237   // we use a timeout here to let the progress bar be able to breath and show
238   // something
239   setTimeout(function() {
240 
241     // check the file format which returns the filepath, extension and the
242     // parser
243     var _checkresult = this.checkFileFormat(container);
244     var parser = _checkresult[2]; // the (uninstantiated) parser
245     var flags = _checkresult[3]; // some (optional) custom flags
246     
247     // instantiate the parser
248     var _parser = new parser;
249     
250     // listen once to a modified event
251     goog.events.listenOnce(_parser, X.event.events.MODIFIED, this.complete
252         .bind(this));
253     
254     // call the parse function and pass in the container, the object and the
255     // data stream and some additional value
256     _parser.parse(container, object, request.response, flags);
257     
258   }.bind(this), 100);
259   
260 };
261 
262 
263 /**
264  * The complete callback which gets executed after successful parsing of a data
265  * stream. A X.event.ModifiedEvent is fired to inform the X.renderer that the
266  * loading was completed.
267  * 
268  * @param {!X.event.ModifiedEvent} event The modified event holding container
269  *          and object.
270  */
271 X.loader.prototype.complete = function(event) {
272 
273   // parsing completed, add progress
274   this.addProgress(1.0);
275   
276   // we use a timeout here to let the progress bar be able to breath and show
277   // something
278   setTimeout(function() {
279 
280     var container = event._container;
281     var object = event._object;
282     
283     // mark the container's file as clean
284     container._file._dirty = false;
285     
286     // fire the modified event on the object
287     object.modified();
288     
289     // mark the loading job as completed
290     this._jobs.set(container._id, true);
291     
292   }.bind(this), 100);
293   
294 };

297 /**
298  * The failure callback which gets executed if anything goes wrong during
299  * downloading. It always throws an error.
300  * 
301  * @param {!XMLHttpRequest} request The original XHR.
302  * @param {!X.base} container The container which has a X.file() attached.
303  * @param {!X.object} object The X.object which is the parent of the container
304  *          or equals it.
305  * @throws {Error} An error, stating that something went wrong.
306  */
307 X.loader.prototype.failed = function(request, container, object) {
308 
309   throw new Error('Loading failed: ', container, object);
310   
311 };

314 /**
315  * Supported data types by extension.
316  * 
317  * @enum {Array}
318  */
319 X.loader.extensions = {
320   // support for the following extensions and the mapping to X.parsers as well
321   // as some custom flags and the result type
322   'STL': [X.parserSTL, null, null],
323   'VTK': [X.parserVTK, null, null],
324   'TRK': [X.parserTRK, null, null],
325   // FSM, INFLATED, SMOOTHWM, SPHERE, PIAL and ORIG are all freesurfer meshes
326   'FSM': [X.parserFSM, null, null],
327   'INFLATED': [X.parserFSM, null, null],
328   'SMOOTHWM': [X.parserFSM, null, null],
329   'SPHERE': [X.parserFSM, null, null],
330   'PIAL': [X.parserFSM, null, null],
331   'ORIG': [X.parserFSM, null, null],
332   'NRRD': [X.parserNRRD, null, null],
333   'CRV': [X.parserCRV, null, null],
334   'MGH': [X.parserMGZ, false, null],
335   'MGZ': [X.parserMGZ, true, null],
336   'TXT': [X.parserLUT, null, null],
337   'LUT': [X.parserLUT, null, null],
338   'PNG': [X.parserIMAGE, 'png', 'arraybuffer'], // here we use the arraybuffer
339   // response type
340   'JPG': [X.parserIMAGE, 'jpeg', 'arraybuffer'],
341   'JPEG': [X.parserIMAGE, 'jpeg', 'arraybuffer'],
342   'GIF': [X.parserIMAGE, 'gif', 'arraybuffer']
343 };
