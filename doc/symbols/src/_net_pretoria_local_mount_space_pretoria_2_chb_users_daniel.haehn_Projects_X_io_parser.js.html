 84 /**
 85  * Process a numerical array and calculate some basic stats: o mean o variance o
 86  * deviation o prod o sum o min; minIndex o max; maxIndex
 87  * 
 88  * @param {*} data The numerical data array to process.
 89  * @return {*} The results.
 90  */
 91 X.parser.prototype.stats_calc = function(data) {
 92 
 93   var r = {
 94     mean: 0,
 95     variance: 0,
 96     deviation: 0,
 97     prod: 1,
 98     sum: 0,
 99     min: 0,
100     minIndex: 0,
101     max: 0,
102     maxIndex: 0
103   };
104   var t = data.length;
105   r.size = t;
106   for ( var m = 0, p = 1, s = 0, l = t; l >= 0; l--) {
107     s += data[l];
108     p *= data[l];
109     if (r.min >= data[l]) {
110       r.min = data[l];
111       r.minIndex = l;
112     }
113     if (r.max <= data[l]) {
114       r.max = data[l];
115       r.maxIndex = l;
116     }
117   }
118   ;
119   r.prod = p;
120   r.sum = s;
121   for (m = r.mean = s / t, l = t, s = 0; l--; s += Math.pow(data[l] - m, 2)) {
122     ;
123   }
124   return r.deviation = Math.sqrt(r.variance = s / t), r;
125 };

148 /**
149  * Parse a Float32 value from some data.
150  * 
151  * @param {!String} data The data to parse.
152  * @param {!number} offset An offset index.
153  * @return {!number} The parsed value.
154  */
155 X.parser.prototype.parseFloat32 = function(data, offset) {
156 
157   var b3 = this.parseUChar8(data, offset), b2 = this.parseUChar8(data,
158       offset + 1), b1 = this.parseUChar8(data, offset + 2), b0 = this
159       .parseUChar8(data, offset + 3),
160 
161   sign = 1 - (2 * (b0 >> 7)), exponent = (((b0 << 1) & 0xff) | (b1 >> 7)) - 127, mantissa = ((b1 & 0x7f) << 16) |
162       (b2 << 8) | b3;
163   
164   if (mantissa == 0 && exponent == -127) {
165     return 0.0;
166   }
167   
168   return sign * (1 + mantissa * Math.pow(2, -23)) * Math.pow(2, exponent);
169   
170 };

198 /**
199  * Parse an array of Float32 values from some data.
200  * 
201  * @param {!String} data The data to parse.
202  * @param {!number} offset An offset index.
203  * @param {!number} elements The number of elements.
204  * @return {!Array} An array consisting of [the actual values as an array, the
205  *         max value, the min value].
206  */
207 X.parser.prototype.parseFloat32Array = function(data, offset, elements) {
208 
209   var arr = new Array();
210   
211   var max = 0;
212   var min = Infinity;
213   
214   var i;
215   for (i = 0; i < elements; i++) {
216     var val = this.parseFloat32(data, offset + (i * 4));
217     arr[i] = val;
218     max = Math.max(max, val);
219     min = Math.min(min, val);
220   }
221   
222   return [arr, max, min];
223 };

482 /**
483  * Reslice a data stream to fill the slices of an X.volume in X,Y and Z
484  * directions. The given volume (object) has to be created at this point
485  * according to the proper dimensions. This also takes care of a possible
486  * associated label map which has to be loaded before.
487  * 
488  * @param {!X.object} object The X.volume to fill.
489  * @param {!Array} datastream The datastream as an array.
490  * @param {!Array} sizes The sizes of the volume as an array [X,Y,Z].
491  * @param {!number} min The min. scalar intensity value.
492  * @param {!number} max The max. scalar intensity value.
493  */
494 X.parser.prototype.reslice = function(object, datastream, sizes, min, max) {
495 
496   // number of slices in scan direction
497   var slices = sizes[2];
498   // number of rows in each slice in scan direction
499   var rowsCount = sizes[1];
500   // number of cols in each slice in scan direction
501   var colsCount = sizes[0];
502   
503   // do we have a labelmap?
504   var hasLabelMap = object._labelmap != null;
505   
506   // slice dimensions in scan direction
507   var numberPixelsPerSlice = rowsCount * colsCount;
508   
509   // allocate 3d image array [slices][rows][cols]
510   var image = new Array(slices);
511   for ( var iS = 0; iS < slices; iS++) {
512     image[iS] = new Array(rowsCount);
513     for ( var iR = 0; iR < rowsCount; iR++) {
514       image[iS][iR] = new Array(colsCount);
515     }
516   }
517   
518   var pixelValue = 0;
519   
520   // loop through all slices in scan direction
521   //
522   // this step creates the slices in X-direction and fills the 3d image array at
523   // the same time
524   // combining the two operations saves some time..
525   var z = 0;
526   for (z = 0; z < slices; z++) {
527     
528     // grab the pixels for the current slice z
529     var currentSlice = datastream.slice(z * (numberPixelsPerSlice), (z + 1) *
530         numberPixelsPerSlice);
531     // the texture has 3 times the pixel value + 1 opacity value for all pixels
532     var textureForCurrentSlice = new Uint8Array(4 * numberPixelsPerSlice);
533     
534     // now loop through all pixels of the current slice
535     var row = 0;
536     var col = 0;
537     var p = 0; // just a counter
538     
539     for (row = 0; row < rowsCount; row++) {
540       for (col = 0; col < colsCount; col++) {
541         
542         // map pixel values
543         pixelValue = currentSlice[p];
544         var pixelValue_r = 0;
545         var pixelValue_g = 0;
546         var pixelValue_b = 0;
547         var pixelValue_a = 0;
548         if (object._colortable) {
549           // color table!
550           var lookupValue = object._colortable._map.get(Math.floor(pixelValue));
551           
552           // check for out of range and use the last label value in this case
553           if (!lookupValue) {
554             lookupValue = object._colortable._map.get(object._colortable._map
555                 .getCount() - 1);
556           }
557           
558           pixelValue_r = 255 * lookupValue[1];
559           pixelValue_g = 255 * lookupValue[2];
560           pixelValue_b = 255 * lookupValue[3];
561           pixelValue_a = 255 * lookupValue[4];
562         } else {
563           // no color table, 1-channel gray value
564           pixelValue_r = pixelValue_g = pixelValue_b = 255 * (pixelValue / max);
565           pixelValue_a = 255;
566         }
567         
568         var textureStartIndex = p * 4;
569         textureForCurrentSlice[textureStartIndex] = pixelValue_r;
570         textureForCurrentSlice[++textureStartIndex] = pixelValue_g;
571         textureForCurrentSlice[++textureStartIndex] = pixelValue_b;
572         textureForCurrentSlice[++textureStartIndex] = pixelValue_a;
573         
574         // save the pixelValue in the 3d image data
575         image[z][row][col] = pixelValue;
576         
577         p++;
578         
579       }
580       
581     }
582     
583     // create the texture for slices in X-direction
584     var pixelTexture = new X.texture();
585     pixelTexture._rawData = textureForCurrentSlice;
586     pixelTexture._rawDataWidth = colsCount;
587     pixelTexture._rawDataHeight = rowsCount;
588     
589     currentSlice = object._slicesZ._children[z];
590     currentSlice._texture = pixelTexture;
591     if (hasLabelMap) {
592       
593       // if this object has a labelmap,
594       // we have it loaded at this point (for sure)
595       // ..so we can attach it as the second texture to this slice
596       currentSlice._labelmap = object._labelmap._slicesZ._children[z]._texture;
597       
598     }
599     
600   }
601   
602   // the following parses the 3d image array according to the Y- and the
603   // Z-direction of the slices
604   // this was unrolled for more performance
605   
606   // for Y-direction
607   // all slices are along the rows of the image
608   // all rows are along the slices of the image
609   // all cols are along the cols of the image
610   //  
611   for (row = 0; row < rowsCount; row++) {
612     
613     var textureForCurrentSlice = new Uint8Array(4 * slices * colsCount);
614     var p = 0; // just a counter
615     for (z = 0; z < slices; z++) {
616       for (col = 0; col < colsCount; col++) {
617         
618         pixelValue = image[z][row][col];
619         var pixelValue_r = 0;
620         var pixelValue_g = 0;
621         var pixelValue_b = 0;
622         var pixelValue_a = 0;
623         if (object._colortable) {
624           // color table!
625           var lookupValue = object._colortable._map.get(Math.floor(pixelValue));
626           
627           // check for out of range and use the last label value in this case
628           if (!lookupValue) {
629             lookupValue = object._colortable._map.get(object._colortable._map
630                 .getCount() - 1);
631           }
632           
633           pixelValue_r = 255 * lookupValue[1];
634           pixelValue_g = 255 * lookupValue[2];
635           pixelValue_b = 255 * lookupValue[3];
636           pixelValue_a = 255 * lookupValue[4];
637         } else {
638           // no color table, 1-channel gray value
639           pixelValue_r = pixelValue_g = pixelValue_b = 255 * (pixelValue / max);
640           pixelValue_a = 255;
641         }
642         
643         var textureStartIndex = p * 4;
644         textureForCurrentSlice[textureStartIndex] = pixelValue_r;
645         textureForCurrentSlice[++textureStartIndex] = pixelValue_g;
646         textureForCurrentSlice[++textureStartIndex] = pixelValue_b;
647         textureForCurrentSlice[++textureStartIndex] = pixelValue_a;
648         
649         p++;
650         
651       }
652     }
653     
654     var pixelTexture = new X.texture();
655     pixelTexture._rawData = textureForCurrentSlice;
656     pixelTexture._rawDataWidth = colsCount;
657     pixelTexture._rawDataHeight = slices;
658     
659     currentSlice = object._slicesY._children[row];
660     currentSlice._texture = pixelTexture;
661     if (hasLabelMap) {
662       
663       // if this object has a labelmap,
664       // we have it loaded at this point (for sure)
665       // ..so we can attach it as the second texture to this slice
666       currentSlice._labelmap = object._labelmap._slicesY._children[row]._texture;
667       
668     }
669     
670   }
671   
672 
673   // for Z
674   // all slices are along the cols of the image
675   // all rows are along the slices of the image
676   // all cols are along the rows of the image
677   //  
678   for (col = 0; col < colsCount; col++) {
679     var textureForCurrentSlice = new Uint8Array(4 * slices * rowsCount);
680     var p = 0; // just a counter
681     for (z = 0; z < slices; z++) {
682       for (row = 0; row < rowsCount; row++) {
683         
684         pixelValue = image[z][row][col];
685         var pixelValue_r = 0;
686         var pixelValue_g = 0;
687         var pixelValue_b = 0;
688         var pixelValue_a = 0;
689         if (object._colortable) {
690           // color table!
691           var lookupValue = object._colortable._map.get(Math.floor(pixelValue));
692           
693           // check for out of range and use the last label value in this case
694           if (!lookupValue) {
695             lookupValue = object._colortable._map.get(object._colortable._map
696                 .getCount() - 1);
697           }
698           
699           pixelValue_r = 255 * lookupValue[1];
700           pixelValue_g = 255 * lookupValue[2];
701           pixelValue_b = 255 * lookupValue[3];
702           pixelValue_a = 255 * lookupValue[4];
703         } else {
704           // no color table, 1-channel gray value
705           pixelValue_r = pixelValue_g = pixelValue_b = 255 * (pixelValue / max);
706           pixelValue_a = 255;
707         }
708         
709         var textureStartIndex = p * 4;
710         textureForCurrentSlice[textureStartIndex] = pixelValue_r;
711         textureForCurrentSlice[++textureStartIndex] = pixelValue_g;
712         textureForCurrentSlice[++textureStartIndex] = pixelValue_b;
713         textureForCurrentSlice[++textureStartIndex] = pixelValue_a;
714         
715         p++;
