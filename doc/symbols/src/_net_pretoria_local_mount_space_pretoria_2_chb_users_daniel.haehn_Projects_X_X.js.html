 56  * The counter class, keeping track of instance ids.

 58  * @constructor

 61 
 62   this._counter = 0;

 65   /**

 70   this.uniqueId = function() {

 82 /**
 83  * Injection mechanism for mixins (from
 84  * http://ejohn.org/blog/javascript-getters-and-setters/) which means copying
 85  * properties, getters/setters and functions from a source object to a target.
 86  * Works best on instances.
 87  * 
 88  * @param {Object} a The target object.
 89  * @param {Object} b The source object.
 90  * @return {Object} The altered object.
 91  */
 92 function inject(a, b) {
 93 
 94   for ( var i in b) { // iterate over all properties
 95     // get getter and setter functions
 96     var g = b.__lookupGetter__(i), s = b.__lookupSetter__(i);
 97     
 98     if (g || s) { // if there is a getter or setter
 99       if (g) {
100         a.__defineGetter__(i, g); // copy getter to new object
101       }
102       if (s) {
103         a.__defineSetter__(i, s); // copy setter to new object
104       }
105     } else {
106       a[i] = b[i]; // just copy the value; nothing special
107     }
108   }
109   return a; // return the altered object
110 }

116 //
117 // 1. Safari does not support the .bind(this) functionality which is crucial for

120 if (!Function.prototype.bind) {
121   
122   Function.prototype.bind = function(oThis) {
123 
124     if (typeof this !== "function") {
125       // closest thing possible to the ECMAScript 5 internal IsCallable function
126       throw new TypeError(
127           "Function.prototype.bind - what is trying to be bound is not callable");
128     }
129     
130     var fSlice = Array.prototype.slice, aArgs = fSlice.call(arguments, 1), fToBind = this;
131     
132     /**
133      * @constructor
134      */
135     var fNOP = function() {
136 
137     };
138     
139     var fBound = function() {
140 
141       return fToBind.apply(this instanceof fNOP ? this : oThis || window, aArgs
142           .concat(fSlice.call(arguments)));
143     };
144     
145     fNOP.prototype = this.prototype;
146     fBound.prototype = new fNOP();
147     
148     return fBound;
149   };
150 }
