 31 goog.provide('X.caption');

 33 // requires
 34 goog.require('X.base');
 35 goog.require('X.interactor');
 36 goog.require('goog.dom');
 37 goog.require('goog.positioning.ViewportPosition');
 38 goog.require('goog.ui.Tooltip');

 41 /**
 42  * Create a caption element (similar to a tooltip).
 43  * 
 44  * @constructor
 45  * @param {!Element} parent The parent element in the DOM tree.
 46  * @param {!number} x The X-coordinate of the upper left caption corner.
 47  * @param {!number} y The Y-coordinate of the upper left caption corner.
 48  * @param {?X.interactor} interactor The interactor for listening to the
 49  *          X.event.HoverEndEvent.
 50  * @extends goog.ui.Tooltip
 51  */
 52 X.caption = function(parent, x, y, interactor) {
 53 
 54   // check if we have a valid parent
 55   if (!goog.isDefAndNotNull(parent)) {
 56     
 57     throw new Error('No valid parent element.');
 58     
 59   }
 60   
 61   // check if we have valid coordinates
 62   if (!goog.isNumber(x) || !goog.isNumber(y)) {
 63     
 64     throw new Error('Invalid coordinates.');
 65     
 66   }
 67   
 68   // check if we have a valid interactor
 69   if (!goog.isDefAndNotNull(interactor) ||
 70       !(interactor instanceof X.interactor)) {
 71     
 72     throw new Error('Invalid interactor.');
 73     
 74   }
 75   
 76   //
 77   // call the standard constructor of goog.ui.Tooltip
 78   goog.base(this);
 79   
 80   //
 81   // class attributes
 82   
 83   /**
 84    * The className of this class.
 85    * 
 86    * @type {string}
 87    * @protected
 88    */
 89   this._classname = 'caption';
 90   
 91   /**
 92    * The parent element in the DOM tree of this caption element.
 93    * 
 94    * @type {!Element}
 95    * @protected
 96    */
 97   this._parent = parent;
 98   
 99   /**
100    * The X-coordinate of this caption element.
101    * 
102    * @type {!number}
103    * @protected
104    */
105   this._x = x;
106   
107   /**
108    * The Y-coordinate of this caption element.
109    * 
110    * @type {!number}
111    * @protected
112    */
113   this._y = y;
114   
115   /**
116    * The interactor for listening to the X.event.HoverEndEvent.
117    * 
118    * @type {?X.interactor}
119    * @protected
120    */
121   this._interactor = interactor;
122   
123   /**
124    * The element for the CSS style of this caption element.
125    * 
126    * @type {?Element}
127    * @protected
128    */
129   this._style = null;
130   
131   /**
132    * The collection of CSS definitions.
133    * 
134    * @type {!Array}
135    * @protected
136    */
137   this._css = [];
138   // configure some styles
139   var css1 = '.x-tooltip {\n';
140   css1 += '  background: #C0C0FF;\n';
141   css1 += '  color: #000000;\n';
142   css1 += '  border: 1px solid infotext;\n';
143   css1 += '  padding: 1px;\n';
144   css1 += '  font-family: sans-serif;\n';
145   // css1 += ' width: 120px;\n';
146   css1 += '}';
147   this._css = [css1];
148   
149   // listen to the X.event.HoverEndEvent from the interactor and kill this
150   // caption element, if caught
151   goog.events.listenOnce(interactor, X.event.events.HOVER_END, this.kill
152       .bind(this));
153   
154   // now initialize
155   this.init_();
156   
157 };
158 // inherit from X.base
159 goog.inherits(X.caption, goog.ui.Tooltip);
160 
161 /**
162  * Internal function to initialize the caption element which gets called by the
163  * constructor.
164  * 
165  * @private
166  */
167 X.caption.prototype.init_ = function() {
168 
169   // enable relative positioning for the main container
170   // this is required to place the progressBar in the center
171   
172   if (this._parent.style.position == 'static' ||
173       this._parent.style.position == '') {
174     this._parent.style.position = 'relative';
175   }
176   
177   //
178   // apply CSS styles to the document
179   //
180   var head = goog.dom.getDocument().getElementsByTagName('head')[0];
181   var style = goog.dom.createDom('style');
182   style.type = 'text/css';
183   style.media = 'screen';
184   var css = goog.dom.createTextNode(this._css[0]);
185   goog.dom.appendChild(head, style);
186   goog.dom.appendChild(style, css);
187   // save the style element for later
188   this._style = style;
189   
190   this.setPosition(new goog.positioning.ViewportPosition(this._x, this._y));
191   this.setVisible(true);
192   
193   // ..render
194   this.attach(this._parent);
195   
196   // get the actual DOM element of this caption element
197   var captionElement = this.getElement();
198   
199   // .. change the color to green
200   captionElement.classList.add('x-tooltip');
201   
202 };
203 
204 
205 /**
206  * Remove the caption element from the document.
207  */
208 X.caption.prototype.kill = function() {
209 
210   // hide the caption element
211   
212   this.setVisible(false);
213   
214   if (this._style) {
215     goog.dom.removeNode(this._style);
216   }
217   if (this.getElement()) {
218     goog.dom.removeNode(this.getElement());
219   }
220   
221   this._style = null;
222   
223 };
