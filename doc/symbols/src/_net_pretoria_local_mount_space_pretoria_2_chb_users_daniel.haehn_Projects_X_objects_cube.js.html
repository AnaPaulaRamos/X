 41 /**
 42  * Create a displayable cube.
 43  * 
 44  * @constructor
 45  * @extends X.object
 46  * @borrows X.constructable#union as #union
 47  * @borrows X.constructable#inverse as #inverse
 48  * @borrows X.constructable#intersect as #intersect
 49  * @borrows X.constructable#subtract as #subtract
 50  */
 51 X.cube = function() {
 52 
 53   //
 54   // call the standard constructor of X.base
 55   goog.base(this);
 56   
 57   //
 58   // class attributes
 59   
 60   /**
 61    * @inheritDoc
 62    * @const
 63    */
 64   this._classname = 'cube';
 65   
 66   /**
 67    * The center of this cube in the world space.
 68    * 
 69    * @type {!Array}
 70    * @protected
 71    */
 72   this._center = [0, 0, 0];
 73   
 74   /**
 75    * The edge length in X-direction.
 76    * 
 77    * @type {!number}
 78    * @protected
 79    */
 80   this._lengthX = 20;
 81   
 82   /**
 83    * The edge length in Y-direction.
 84    * 
 85    * @type {!number}
 86    * @protected
 87    */
 88   this._lengthY = 20;
 89   
 90   /**
 91    * The edge length in Z-direction.
 92    * 
 93    * @type {!number}
 94    * @protected
 95    */
 96   this._lengthZ = 20;
 97   
 98   /**
 99    * @inheritDoc
100    * @const
101    */
102   this._textureCoordinateMap = [
103 
104   // Right face
105   0, 1, 1, 1, 1, 0, 0, 1, 1, 0, 0, 0,
106 
107   // Left face
108   1, 1, 1, 0, 0, 0, 1, 1, 0, 0, 0, 1,
109 
110   // Bottom face
111   0, 1, 1, 1, 1, 0, 0, 1, 1, 0, 0, 0,
112 
113   // Top face
114   0, 0, 0, 1, 1, 1, 0, 0, 1, 1, 1, 0,
115 
116   // Back face
117   1, 1, 1, 0, 0, 0, 1, 1, 0, 0, 0, 1,
118 
119   // Front face
120   0, 1, 1, 1, 1, 0, 0, 1, 1, 0, 0, 0
121 
122   ];
123   
124   inject(this, new X.constructable()); // this object is constructable
125   
126 };
127 // inherit from X.object
128 goog.inherits(X.cube, X.object);

131 /**
132  * Get the center of this cube.
133  * 
134  * @return {!Array} The center as an array with length 3.
135  * @public
136  */
137 X.cube.prototype.__defineGetter__('center', function() {
138 
139   return this._center;
140   
141 });
142 
143 
144 /**
145  * Set the center of this cube.
146  * 
147  * @param {!Array} center The center as an array with length 3 ([X,Y,Z]).
148  * @throws {Error} An error, if the center is invalid.
149  * @public
150  */
151 X.cube.prototype.__defineSetter__('center', function(center) {
152 
153   if (!goog.isDefAndNotNull(center) || !(center instanceof Array) ||
154       (center.length != 3)) {
155     
156     throw new Error('Invalid center');
157     
158   }
159   
160   this._center = center;
161   
162 });

264 /**
265  * @inheritDoc
266  */
267 X.cube.prototype.modified = function() {
268 
269   this.fromCSG(new CSG.cube({
270     center: this._center,
271     radius: [this._lengthX / 2, this._lengthY / 2, this._lengthZ / 2]
272   }));
273   
274   // call the super class
275   goog.base(this, 'modified');
276   
277 };
