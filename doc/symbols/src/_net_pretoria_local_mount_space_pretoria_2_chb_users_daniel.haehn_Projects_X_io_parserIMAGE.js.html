 30 // provides
 31 goog.provide('X.parserIMAGE');
 32 
 33 // requires
 34 goog.require('X.event');
 35 goog.require('X.parser');
 36 
 37 
 38 
 39 /**
 40  * Create a parser for images (PNG/JPG/JPEG/GIF).
 41  * 
 42  * @constructor
 43  * @extends X.parser
 44  */
 45 X.parserIMAGE = function() {
 46 
 47   //
 48   // call the standard constructor of X.base
 49   goog.base(this);
 50   
 51   //
 52   // class attributes
 53   
 54   /**
 55    * @inheritDoc
 56    * @const
 57    */
 58   this._classname = 'parserIMAGE';
 59   
 60 };
 61 // inherit from X.parser
 62 goog.inherits(X.parserIMAGE, X.parser);
 63 
 64 
 65 /**
 66  * @inheritDoc
 67  */
 68 X.parserIMAGE.prototype.parse = function(container, object, data, flag) {
 69 
 70   if (!(data instanceof ArrayBuffer)) {
 71     
 72     throw new Error();
 73     
 74   }
 75   
 76   // convert data to a byte array
 77   var bytebuffer = new Uint8Array(data);
 78   var i = bytebuffer.length;
 79   
 80   // create a binary string of the bytebuffer
 81   var binaryString = new Array(i);
 82   while (i--) {
 83     binaryString[i] = String.fromCharCode(bytebuffer[i]);
 84   }
 85   var convertedData = binaryString.join('');
 86   
 87   // encode the converted binary string
 88   var encodedData = window.btoa(convertedData);
 89   
 90   // create a new image
 91   var _image = new Image();
 92   
 93   // we need to wait until the image is properly set up
 94   goog.events.listenOnce(_image, goog.events.EventType.LOAD,
 95       this.parseCompleted.bind(this, _image, container, object, data, flag));
 96   
 97   // set the encoded data using a data uri
 98   _image.src = "data:image/" + flag + ";base64," + encodedData;
 99   
100 };
101 
102 
103 /**
104  * The callback which gets called when the image was set up using data uri. We
105  * fire the modified event here to tell the X.loader that we are done.
106  * 
107  * @param {!Image} image The image container.
108  * @param {!X.base} container A container which holds the loaded data. This can
109  *          be an X.object as well.
110  * @param {!X.object} object The object to configure.
111  * @param {!String} data The data to parse.
112  * @param {*} flag An additional flag.
113  */
114 X.parserIMAGE.prototype.parseCompleted = function(image, container, object,
115     data, flag) {
116 
117   // attach the image to the container
118   container._image = image;
119   
120   // the object should be set up here, so let's fire a modified event
121   var modifiedEvent = new X.event.ModifiedEvent();
122   modifiedEvent._object = object;
123   modifiedEvent._container = container;
124   this.dispatchEvent(modifiedEvent);
125   
126 };

128 // export symbols (required for advanced compilation)
129 goog.exportSymbol('X.parserIMAGE', X.parserIMAGE);
130 goog.exportSymbol('X.parserIMAGE.prototype.parse',
131     X.parserIMAGE.prototype.parse);
