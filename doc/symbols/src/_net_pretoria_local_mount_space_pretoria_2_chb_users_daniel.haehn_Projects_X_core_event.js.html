 53 /**
 54  * The super class for all events in XTK.
 55  * 
 56  * @constructor
 57  * @param {string} type A type identifier for this event.
 58  * @extends goog.events.Event
 59  */
 60 X.event = function(type) {
 61 
 62   //
 63   // call the default event constructor
 64   goog.base(this, type);
 65   
 66   //
 67   // class attributes
 68   
 69   /**
 70    * The className of this class.
 71    * 
 72    * @type {string}
 73    * @protected
 74    */
 75   this._classname = 'event';
 76   
 77 };
 78 // inherit from goog.events.Event
 79 goog.inherits(X.event, goog.events.Event);

 82 /**
 83  * Creates a unique event id.
 84  * 
 85  * @param {string} id The id.
 86  * @return {string} A unique id.
 87  */
 88 X.event.uniqueId = function(id) {
 89 
 90   return goog.events.getUniqueId(id);
 91   
 92 };

 94 /**
 95  * The events of this class.
 96  * 
 97  * @enum {string}
 98  */
 99 X.event.events = {
100   // the pan event, where the event and focus get moved accordingly
101   PAN: X.event.uniqueId('pan'),
102   
103   // the rotate event, where only the event gets moved
104   ROTATE: X.event.uniqueId('rotate'),
105   
106   // the zoom event, where the event Z coordinate changes
107   ZOOM: X.event.uniqueId('zoom'),
108   
109   // the scroll event
110   SCROLL: X.event.uniqueId('scroll'),
111   
112   // the render event
113   RENDER: X.event.uniqueId('render'),
114   
115   // the reset view event
116   RESETVIEW: X.event.uniqueId('resetview'),
117   
118   // the object modified event
119   MODIFIED: X.event.uniqueId('modified'),
120   
121   // the loading progress event
122   PROGRESS: X.event.uniqueId('progress'),
123   
124   // the hover event
125   HOVER: X.event.uniqueId('hover'),
126   
127   // the hover end event
128   HOVER_END: X.event.uniqueId('hover_end')
129 
130 };

181 /**
182  * The zoom event to initiate zoom in or zoom out.
183  * 
184  * @constructor
185  * @extends X.event
186  */
187 X.event.ZoomEvent = function() {
188 
189   // call the default event constructor
190   goog.base(this, X.event.events.ZOOM);
191   
192   /**
193    * The flag for the zooming direction. If TRUE, the zoom operation will move
194    * the objects closer to the event. If FALSE, further away from the event.
195    * 
196    * @type {!boolean}
197    * @protected
198    */
199   this._in = false;
200   
201   /**
202    * The flag for the zooming speed. If TRUE, the zoom operation will happen
203    * fast. If FALSE, there will be a fine zoom operation.
204    * 
205    * @type {!boolean}
206    * @protected
207    */
208   this._fast = false;
209   
210 };
211 // inherit from goog.events.Event
212 goog.inherits(X.event.ZoomEvent, X.event);
