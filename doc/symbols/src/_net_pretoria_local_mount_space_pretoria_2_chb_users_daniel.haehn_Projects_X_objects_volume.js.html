 40 /**
 41  * Create a displayable volume which consists of X.slices in X,Y and Z direction
 42  * and can also be volume rendered.
 43  * 
 44  * @constructor
 45  * @param {X.volume=} volume Another X.volume to use as a template.
 46  * @extends X.object
 47  */
 48 X.volume = function(volume) {
 49 
 50   //
 51   // call the standard constructor of X.base
 52   goog.base(this);
 53   
 54   //
 55   // class attributes
 56   
 57   /**
 58    * @inheritDoc
 59    * @const
 60    */
 61   this._classname = 'volume';
 62   
 63   /**
 64    * The center of this volume.
 65    * 
 66    * @type {!Array}
 67    * @protected
 68    */
 69   this._center = [0, 0, 0];
 70   
 71   /**
 72    * The dimensions of this volume.
 73    * 
 74    * @type {!Array}
 75    * @protected
 76    */
 77   this._dimensions = [10, 10, 10];
 78   
 79   /**
 80    * The spacing of this volume.
 81    * 
 82    * @type {!Array}
 83    * @protected
 84    */
 85   this._spacing = [1, 1, 1];
 86   
 87   /**
 88    * The index of the currently shown slice in X-direction.
 89    * 
 90    * @type {!number}
 91    * @public
 92    */
 93   this._indexX = 0;
 94   
 95   /**
 96    * The index of the formerly shown slice in X-direction.
 97    * 
 98    * @type {!number}
 99    * @protected
100    */
101   this._indexXold = 0;
102   
103   /**
104    * The index of the currently shown slice in Y-direction.
105    * 
106    * @type {!number}
107    * @public
108    */
109   this._indexY = 0;
110   
111   /**
112    * The index of the formerly shown slice in Y-direction.
113    * 
114    * @type {!number}
115    * @protected
116    */
117   this._indexYold = 0;
118   
119   /**
120    * The index of the currently shown slice in Z-direction.
121    * 
122    * @type {!number}
123    * @public
124    */
125   this._indexZ = 0;
126   
127   /**
128    * The index of the formerly shown slice in Z-direction.
129    * 
130    * @type {!number}
131    * @protected
132    */
133   this._indexZold = 0;
134   
135   /**
136    * The X.object holding the slices in X-direction.
137    * 
138    * @type {!X.object}
139    * @protected
140    */
141   this._slicesX = new X.object();
142   
143   /**
144    * The X.object holding the slices in Y-direction.
145    * 
146    * @type {!X.object}
147    * @protected
148    */
149   this._slicesY = new X.object();
150   
151   /**
152    * The X.object holding the slices in Z-direction.
153    * 
154    * @type {!X.object}
155    * @protected
156    */
157   this._slicesZ = new X.object();
158   
159   /**
160    * The toggle for volume rendering or cross-sectional slicing.
161    * 
162    * @type {boolean}
163    * @public
164    */
165   this._volumeRendering = false;
166   
167   /**
168    * The cached toggle for volume rendering or cross-sectional slicing.
169    * 
170    * @type {boolean}
171    * @public
172    */
173   this._volumeRenderingOld = false;
174   
175   /**
176    * The direction for the volume rendering. This is used for caching.
177    * 
178    * @type {!number}
179    * @private
180    */
181   this._volumeRenderingDirection = 0;
182   
183   /**
184    * The label map of this volume.
185    * 
186    * @type {?X.volume}
187    * @private
188    */
189   this._labelmap = null;
190   
191   /**
192    * Flag to show borders or not.
193    * 
194    * @type {boolean}
195    * @protected
196    */
197   this._borders = true;
198   
199   // inject functionality
200   inject(this, new X.loadable()); // this object is loadable from a file
201   inject(this, new X.thresholdable()); // this object is thresholdable
202   
203   if (goog.isDefAndNotNull(volume)) {
204     
205     // copy the properties of the given volume over
206     this.copy_(volume);
207     
208   }
209   
210 };
211 // inherit from X.object
212 goog.inherits(X.volume, X.object);

252 /**
253  * Create the volume.
254  * 
255  * @private
256  */
257 X.volume.prototype.create_ = function() {
258 
259   // remove all old children
260   this._children.length = 0;
261   this._slicesX._children.length = 0;
262   this._slicesY._children.length = 0;
263   this._slicesZ._children.length = 0;
264   
265   // add the new children
266   this._children.push(this._slicesX);
267   this._children.push(this._slicesY);
268   this._children.push(this._slicesZ);
269   
270   //
271   // create the slices
272   var xyz = 0; // 0 for x, 1 for y, 2 for z
273   for (xyz = 0; xyz < 3; xyz++) {
274     
275     var halfDimension = (this._dimensions[xyz] - 1) / 2;
276     
277     var _indexCenter = halfDimension;
278     
279     var i = 0;
280     for (i = 0; i < this._dimensions[xyz]; i++) {
281       
282       var _position = (-halfDimension * this._spacing[xyz]) +
283           (i * this._spacing[xyz]);
284       
285       var _center = new Array([this._center[0] + _position, this._center[1],
286                                this._center[2]], [this._center[0],
287                                                   this._center[1] + _position,
288                                                   this._center[2]],
289           [this._center[0], this._center[1], this._center[2] + _position]);
290       
291       var _front = new Array([1, 0, 0], [0, 1, 0], [0, 0, 1]);
292       var _up = new Array([0, 1, 0], [0, 0, -1], [0, 1, 0]);
293       
294       // the container and indices
295       var slices = this._children[xyz]._children;
296       
297       // dimensions
298       var width = 0;
299       var height = 0;
300       var borderColor = [1, 1, 1];
301       var borders = this._borders;
302       if (xyz == 0) {
303         // for x slices
304         width = this._dimensions[2] * this._spacing[2] - this._spacing[2];
305         height = this._dimensions[1] * this._spacing[1] - this._spacing[1];
306         borderColor = [1, 1, 0];
307       } else if (xyz == 1) {
308         // for y slices
309         width = this._dimensions[0] * this._spacing[0] - this._spacing[0];
310         height = this._dimensions[2] * this._spacing[2] - this._spacing[2];
311         borderColor = [1, 0, 0];
312       } else if (xyz == 2) {
313         // for z slices
314         width = this._dimensions[0] * this._spacing[0] - this._spacing[0];
315         height = this._dimensions[1] * this._spacing[1] - this._spacing[1];
316         borderColor = [0, 1, 0];
317       }
318       
319       // for labelmaps, don't create the borders since this would create them 2x
320       if (goog.isDefAndNotNull(this._volume)) {
321         borders = false;
322       }
323       
324       // .. new slice
325       var _slice = new X.slice();
326       _slice.setup(_center[xyz], _front[xyz], _up[xyz], width, height, borders,
327           borderColor);
328       _slice._volume = this;
329       
330       // only show the middle slice, hide everything else
331       _slice['visible'] = (i == Math.floor(_indexCenter));
332       
333       // attach to all _slices with the correct slice index
334       slices.push(_slice);
335       
336     }
337     
338     // by default, all the 'middle' slices are shown
339     if (xyz == 0) {
340       this._indexX = _indexCenter;
341       this._indexXold = _indexCenter;
342     } else if (xyz == 1) {
343       this._indexY = _indexCenter;
344       this._indexYold = _indexCenter;
345     } else if (xyz == 2) {
346       this._indexZ = _indexCenter;
347       this._indexZold = _indexCenter;
348     }
349   }
350   
351   this._dirty = true;
352   
353 };

355 /**
356  * Re-show the slices or re-activate the volume rendering for this volume.
357  * 
358  * @inheritDoc
359  */
360 X.volume.prototype.modified = function() {
361 
362   // only do this if we already have children aka. the create_() method was
363   // called
364   if (this._children.length > 0) {
365     if (this._volumeRendering != this._volumeRenderingOld) {
366       
367       if (this._volumeRendering) {
368         
369         // first, hide possible slicing slices but only if volume rendering was
370         // just switched on
371         var _sliceX = this._children[0]._children[parseInt(this._indexX, 10)];
372         _sliceX['visible'] = false;
373         var _sliceY = this._children[1]._children[parseInt(this._indexY, 10)];
374         _sliceY['visible'] = false;
375         var _sliceZ = this._children[2]._children[parseInt(this._indexZ, 10)];
376         _sliceZ['visible'] = false;
377         
378       } else {
379         
380         // hide the volume rendering slices
381         var _child = this._children[this._volumeRenderingDirection];
382         _child['visible'] = false;
383         
384       }
385       
386       // switch from slicing to volume rendering or vice versa
387       this._dirty = true;
388       this._volumeRenderingOld = this._volumeRendering;
389       
390     }
391     
392     if (this._volumeRendering) {
393       
394       // prepare volume rendering
395       this.volumeRendering_(this._volumeRenderingDirection);
396       
397     } else {
398       
399       // prepare slicing
400       this.slicing_();
401       
402     }
403   }
404   
405   // call the superclass' modified method
406   goog.base(this, 'modified');
407   
408 };

410 /**
411  * Show the current slices which are set by this._indexX, this._indexY and
412  * this._indexZ and hide all others.
413  */
414 X.volume.prototype.slicing_ = function() {
415 
416   // display the current slices in X,Y and Z direction
417   var xyz = 0; // 0 for x, 1 for y, 2 for z
418   for (xyz = 0; xyz < 3; xyz++) {
419     
420     var _child = this._children[xyz];
421     var currentIndex = 0;
422     var oldIndex = 0;
423     
424     // buffer the old indices
425     if (xyz == 0) {
426       currentIndex = this._indexX;
427       oldIndex = this._indexXold;
428       this._indexXold = this._indexX;
429     } else if (xyz == 1) {
430       currentIndex = this._indexY;
431       oldIndex = this._indexYold;
432       this._indexYold = this._indexY;
433     } else if (xyz == 2) {
434       currentIndex = this._indexZ;
435       oldIndex = this._indexZold;
436       this._indexZold = this._indexZ;
437     }
438     
439     // hide the old slice
440     var _oldSlice = _child._children[parseInt(oldIndex, 10)];
441     _oldSlice['visible'] = false;
442     
443     // show the current slice and also show the borders if they exist by
444     // calling the setter of visible rather than accessing the _visible property
445     var _currentSlice = _child._children[parseInt(currentIndex, 10)];
446     _currentSlice['visible'] = true;
447     _currentSlice._opacity = 1.0;
448     
449   }
450   
451 };

681 /**
682  * Perform volume rendering of this volume along a specific direction. The
683  * direction is important since we show tiled 2d textures along the direction
684  * for a clean rendering experience.
685  * 
686  * @param {number} direction The direction of the volume rendering
687  *          (0==x,1==y,2==z).
688  * @protected
689  */
690 X.volume.prototype.volumeRendering_ = function(direction) {
691 
692   if ((!this._volumeRendering) ||
693       (!this._dirty && direction == this._volumeRenderingDirection)) {
694     
695     // we do not have to do anything
696     return;
697     
698   }
699   
700   // hide old volume rendering slices
701   var _child = this._children[this._volumeRenderingDirection];
702   _child['visible'] = false;
703   
704   // show new volume rendering slices, but don't show the borders
705   _child = this._children[direction];
706   var _numberOfSlices = _child._children.length;
707   var i;
708   for (i = 0; i < _numberOfSlices; i++) {
709     _child._children[i]._visible = true;
710   }
711   // _child['visible'] = true;
712   
713   // store the direction
714   this._volumeRenderingDirection = direction;
715   
716   this._dirty = false;
717   
718 };
