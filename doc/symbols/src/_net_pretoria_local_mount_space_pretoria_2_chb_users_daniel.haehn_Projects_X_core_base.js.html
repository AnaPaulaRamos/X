 30 // provides
 31 goog.provide('X.base');
 32 
 33 // requires
 34 goog.require('X');
 35 goog.require('goog.events');
 36 goog.require('goog.events.EventTarget');
 37 
 38 
 39 
 40 /**
 41  * The superclass for all X.base-objects. All derived objects will be registered
 42  * for event handling.
 43  * 
 44  * @constructor
 45  * @extends goog.events.EventTarget
 46  */
 47 X.base = function() {
 48 
 49   //
 50   // register this class within the event system by calling the superclass
 51   // constructor
 52   goog.base(this);
 53   
 54   //
 55   // class attributes
 56   
 57   /**
 58    * The className of this class.
 59    * 
 60    * @type {string}
 61    * @protected
 62    */
 63   this._classname = 'base';
 64   
 65   /**
 66    * The uniqueId of this instance. Each class instance in XTK has a uniqueId.
 67    * 
 68    * @type {number}
 69    * @protected
 70    */
 71   this._id = window["X.counter"].uniqueId();
 72   
 73 
 74   /**
 75    * The 'dirty' flag of this object.
 76    * 
 77    * @type {boolean}
 78    * @protected
 79    */
 80   this._dirty = false;
 81 };
 82 // enable events
 83 goog.inherits(X.base, goog.events.EventTarget);
 84 
 85 
 86 /**
 87  * Get the className of this object.
 88  * 
 89  * @return {string} The className of this object.
 90  * @public
 91  */
 92 X.base.prototype.__defineGetter__('classname', function() {
 93 
 94   return this._classname;
 95   
 96 });
 97 
 98 
 99 /**
100  * Get the id of this instance.
101  * 
102  * @return {string} The className of this object.
103  * @public
104  */
105 X.base.prototype.__defineGetter__('id', function() {
106 
107   return this._id;
108   
109 });
