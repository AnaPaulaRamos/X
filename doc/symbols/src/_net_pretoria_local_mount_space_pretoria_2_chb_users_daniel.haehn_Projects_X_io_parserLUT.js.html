 40 /**
 41  * Create a parser for color maps (look-up tables).
 42  * 
 43  * @constructor
 44  * @extends X.parser
 45  */
 46 X.parserLUT = function() {
 47 
 48   //
 49   // call the standard constructor of X.base
 50   goog.base(this);
 51   
 52   //
 53   // class attributes
 54   
 55   /**
 56    * @inheritDoc
 57    * @const
 58    */
 59   this._classname = 'parserLUT';
 60   
 61 };
 62 // inherit from X.parser
 63 goog.inherits(X.parserLUT, X.parser);
 64 
 65 
 66 /**
 67  * @inheritDoc
 68  */
 69 X.parserLUT.prototype.parse = function(container, object, data, flag) {
 70 
 71   var colortable = container;
 72   
 73   var dataAsArray = data.split('\n');
 74   
 75   var numberOfLines = dataAsArray.length;
 76   
 77   var i;
 78   for (i = 0; i < numberOfLines; ++i) {
 79     
 80     var line = dataAsArray[i];
 81     
 82     // trim the line
 83     line = line.replace(/^\s+|\s+$/g, '');
 84     
 85     // ignore comments
 86     if (line[0] == '#') {
 87       continue;
 88     }
 89     
 90     // split each line
 91     var lineFields = line.split(' ');
 92     
 93     // filter out multiple blanks
 94     lineFields = lineFields.filter(function(v) {
 95 
 96       return v != '';
 97       
 98     });
 99     
100     // check if we have 6 values
101     if (lineFields.length != 6) {
102       
103       // ignore this line
104       continue;
105       
106     }
107     
108     // here, we have a valid array containing
109     // labelValue, labelName, r, g, b, a
110     
111     // convert r, g, b, a to the range 0..1 and don't forget to make it a number
112     lineFields[2] = parseInt(lineFields[2], 10) / 255; // r
113     lineFields[3] = parseInt(lineFields[3], 10) / 255; // g
114     lineFields[4] = parseInt(lineFields[4], 10) / 255; // b
115     lineFields[5] = parseInt(lineFields[5], 10) / 255; // a
116     
117     // .. push it
118     colortable.add(parseInt(lineFields[0], 10), lineFields[1], lineFields[2],
119         lineFields[3], lineFields[4], lineFields[5], 10);
120     
121   }
122   
123   // the object should be set up here, so let's fire a modified event
124   var modifiedEvent = new X.event.ModifiedEvent();
125   modifiedEvent._object = object;
126   modifiedEvent._container = container;
127   this.dispatchEvent(modifiedEvent);
128   
129 };
