 46 /**
 47  * Create a parser for .NRRD files.
 48  * 
 49  * @constructor
 50  * @extends X.parser
 51  */
 52 X.parserNRRD = function() {
 53 
 54   //
 55   // call the standard constructor of X.parser
 56   goog.base(this);
 57   
 58   //
 59   // class attributes
 60   
 61   /**
 62    * @inheritDoc
 63    * @const
 64    */
 65   this._classname = 'parserNRRD';
 66   
 67 };
 68 // inherit from X.parser
 69 goog.inherits(X.parserNRRD, X.parser);

 72 /**
 73  * @inheritDoc
 74  */
 75 X.parserNRRD.prototype.parse = function(container, object, data, flag) {
 76 
 77   // the position in the file
 78   var position = 0;
 79   
 80   // grab the header
 81   var headerRegexMatch = data.match(/^([\s\S]*?)\r?\n\r?\n/);
 82   position = headerRegexMatch[0].length; // the one _with_ the blank line
 83   var header = headerRegexMatch[1]; // the one without the blank line
 84   
 85   // parse the header
 86   this.parseHeader(header);
 87   
 88   // now we have all kinds of things attached to this reader..
 89   // this was done by M. Lauer
 90   // I don't really like it but it works..
 91   
 92   var _data = 0; // the data without header
 93   
 94   if (this.encoding == 'gzip' || this.encoding == 'gz') {
 95     // we need to decompress the datastream
 96     _data = new JXG.Util.Unzip(data.substr(position)).unzip()[0][0];
 97   } else {
 98     // we can use the data directly
 99     _data = data.substr(position);
100   }
101   
102   var numberOfPixels = this.sizes[0] * this.sizes[1] * this.sizes[2];
103   
104   //
105   // parse the (unzipped) data to a datastream of the correct type
106   //
107   var datastream = new Array(numberOfPixels);
108   
109   // we store the min and max values to be able to convert the values to uint8
110   // later
111   var max = -Infinity;
112   var min = Infinity;
113   
114   var i;
115   for (i = 0; i < numberOfPixels; i++) {
116     // parseFunc was defined by analyzing the nrrd header
117     var pixelValue = this.parseFunc(_data, 0 + (i * this.parseBytes));
118     datastream[i] = pixelValue;
119     max = Math.max(max, pixelValue);
120     min = Math.min(min, pixelValue);
121   }
122   
123   //
124   // we know enough to create the object
125   
126   // origin
127   // var _origin = this['space origin'];
128   // _origin = _origin.slice(1, _origin.length).split(',');
129   // object._center[0] = parseFloat(_origin[0]) - this.sizes[0] / 2;
130   // object._center[1] = parseFloat(_origin[1]) - this.sizes[1] / 2;
131   // object._center[2] = parseFloat(_origin[2]) - this.sizes[2] / 2;
132   
133   // dimensions
134   object._dimensions = [this.sizes[0], this.sizes[1], this.sizes[2]];
135   
136   // spacing
137   var spacingX = new goog.math.Vec3(this.vectors[0][0], this.vectors[0][1],
138       this.vectors[0][2]).magnitude();
139   var spacingY = new goog.math.Vec3(this.vectors[1][0], this.vectors[1][1],
140       this.vectors[1][2]).magnitude();
141   var spacingZ = new goog.math.Vec3(this.vectors[2][0], this.vectors[2][1],
142       this.vectors[2][2]).magnitude();
143   object._spacing = [spacingX, spacingY, spacingZ];
144   
145 
146   // attach the scalar range to the volume
147   object._min = min;
148   object._max = max;
149   // .. and set the default threshold
150   // only if the threshold was not already set
151   if (object._lowerThreshold == -Infinity) {
152     object._lowerThreshold = min;
153   }
154   if (object._upperThreshold == Infinity) {
155     object._upperThreshold = max;
156   }
157   
158   // create the object
159   object.create_();
160   
161   // now we have the values and need to reslice in the 3 orthogonal directions
162   // and create the textures for each slice
163   this.reslice(object, datastream, this.sizes, min, max);
164   
165   // the object should be set up here, so let's fire a modified event
166   var modifiedEvent = new X.event.ModifiedEvent();
167   modifiedEvent._object = object;
168   modifiedEvent._container = container;
169   this.dispatchEvent(modifiedEvent);
170   
171 };

174 /**
175  * Parse a NRRD file header.
176  * 
177  * @param {string} header The NRRD header to parse.
178  * @throws {Error} An error, if the NRRD header is not supported or invalid.
179  */
180 X.parserNRRD.prototype.parseHeader = function(header) {
181 
182   var data, field, fn, i, l, lines, m, _i, _len, _results;
183   lines = header.split(/\r?\n/);
184   for (_i = 0, _len = lines.length; _i < _len; _i++) {
185     l = lines[_i];
186     if (l.match(/NRRD\d+/)) {
187       this.isNrrd = true;
188     } else if (l.match(/^#/)) {
189     } else if (m = l.match(/(.*):(.*)/)) {
190       field = m[1].trim();
191       data = m[2].trim();
192       fn = this.fieldFunctions[field];
193       if (fn) {
194         fn.call(this, data);
195       } else {
196         this[field] = data;
197       }
198     }
199   }
200   if (!this.isNrrd) {
201     throw new Error('Not an NRRD file');
202   }
203   if (this.encoding !== 'raw' && this.encoding !== 'gzip' &&
204       this.encoding !== 'gz') {
205     throw new Error('Only raw or gz/gzip encoding is allowed');
206   }
207   if (!this.vectors) {
208     this.vectors = [new goog.math.Vec3(1, 0, 0), new goog.math.Vec3(0, 1, 0),
209                     new goog.math.Vec3(0, 0, 1)];
210     if (this.spacings) {
211       _results = [];
212       for (i = 0; i <= 2; i++) {
213         _results.push(!isNaN(this.spacings[i]) ? this.vectors[i]
214             .scale(this.spacings[i]) : void 0);
215       }
216       return _results;
217     }
218   }
219 };

222 /**
223  * Functions for parsing the NRRD header fields.
224  * 
225  * @type {Object}
226  */
227 X.parserNRRD.prototype.fieldFunctions = {
228   'type': function(data) {
229 
230     switch (data) {
231     case 'unsigned char':
232     case 'uint8':
233       this.parseFunc = this.parseUChar8;
234       this.parseBytes = 1;
235       break;
236     case 'signed char':
237     case 'int8':
238       this.parseFunc = this.parseSChar8;
239       this.parseBytes = 1;
240       break;
241     case 'short':
242     case 'signed short':
243     case 'unsigned short':
244     case 'short int':
245     case 'int16':
246       this.parseFunc = this.parseUInt16;
247       this.parseBytes = 2;
248       break;
249     case 'int':
250     case 'int32':
251       break;
252     case 'float':
253       this.parseFunc = this.parseFloat32;
254       this.parseBytes = 4;
255       break;
256     default:
257       throw new Error('Only short/int/int8/float data is allowed. Found ' +
258           data);
259     }
260     return this.type = data;
261   },
262   'endian': function(data) {
263 
264     return this.endian = data;
265   },
266   'encoding': function(data) {
267 
268     return this.encoding = data;
269   },
270   'dimension': function(data) {
271 
272     return this.dim = parseInt(data, 10);
273   },
274   'sizes': function(data) {
275 
276     var i;
277     return this.sizes = (function() {
278 
279       var _i, _len, _ref, _results;
280       _ref = data.split(/\s+/);
281       _results = [];
282       for (_i = 0, _len = _ref.length; _i < _len; _i++) {
283         i = _ref[_i];
284         _results.push(parseInt(i, 10));
285       }
286       return _results;
287     })();
288   },
289   'space directions': function(data) {
290 
291     var f, parts, v;
292     parts = data.match(/\(.*?\)/g);
293     return this.vectors = (function() {
294 
295       var _i, _len, _results;
296       _results = [];
297       for (_i = 0, _len = parts.length; _i < _len; _i++) {
298         v = parts[_i];
299         _results.push((function() {
300 
301           var _j, _len2, _ref, _results2;
302           _ref = v.slice(1, -1).split(/,/);
303           _results2 = [];
304           for (_j = 0, _len2 = _ref.length; _j < _len2; _j++) {
305             f = _ref[_j];
306             _results2.push(parseFloat(f));
307           }
308           return _results2;
309         })());
310       }
311       return _results;
312     })();
313   },
314   'spacings': function(data) {
315 
316     var f, parts;
317     parts = data.split(/\s+/);
318     return this.spacings = (function() {
319 
320       var _i, _len, _results;
321       _results = [];
322       for (_i = 0, _len = parts.length; _i < _len; _i++) {
323         f = parts[_i];
324         _results.push(parseFloat(f));
325       }
326       return _results;
327     })();
328   }
329 };
